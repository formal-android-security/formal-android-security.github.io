Program Simulation and Process Simulation

Definition ProcessSimulation (ac ac': AnnoCmd) (t: Tid) (l: Label)
            (pc: Label) (envp envp': Env) (rho: Policys) (envs: Env) (i: Tid) :Prop :=
  ∀ pst pst',
    (∀ sst0, ThreadTerm t rho ac pst sst0) →
    (∀ sst'0, ThreadTerm t rho ac' pst' sst'0) →
    PStateApprox pst pst' l pc envp →
    TSim ac ac' pst pst' t l pc envp' rho envs i.

Definition ProgramSimulation (pe pe': Progexe) (l: Label)
           (pcs: PC_Map) (envps envps': Env_Map) (rho: Policys) (envs: Env) (i: Tid) :Prop :=
  ∀ psts psts',
    PStatesApprox psts psts' l pcs envps →
    ProgTerm rho pe psts →
    ProgTerm rho pe' psts' →
    PEPSTSMod.eqdom pe psts →
    PEPSTSMod.eqdom pe' psts' →
    PSim pe pe' psts psts' l pcs envps' rho envs i.

The Proof of Sound

Lemma ProgramSimulation_implies_ProgramEquivalence:
  ∀ pe pe' i l rho pcs envps envs envps'
         (HProgramSimulation: ProgramSimulation pe pe' l pcs envps envps' rho envs i),
    ProgEqu pe pe' i l rho pcs envps envs.
Proof.
  intros.
  unfold ProgEqu.
  intros sigma psts sst sigma' psts' sst' psts1 ss1 hqs1 lqs1 alpha1 Hsigma Hsigma' HPTerm1 HPTerm2 HstII HstII' HStateApprox HstarParaStep Hpepstseqdom Hpepstseqdom'.
  unfold ProgramSimulation in ×.
  subst.
  destruct HStateApprox as [HPStatesApprox HSStateApprox].
  assert (pstatesII psts rho) as HpstsII. {
    destruct HstII; auto.
  }
  assert (pstatesII psts' rho) as HpstsII'. {
    destruct HstII'; auto.
  }
  
  specialize (HProgramSimulation psts psts' HPStatesApprox HPTerm1 HPTerm2 Hpepstseqdom Hpepstseqdom').
  eapply PSim_Skip_SStateApprox_starParaStep in HstarParaStep; eauto.
  destruct HstarParaStep as [psts2 [sst2 [HstarParaStep HSStateApprox']]].
  destruct sst2 as [[[ss2 hqs2] lqs2] alpha2].
  repeat eapply ex_intro.
  split; eauto.
  destruct HSStateApprox' as [HSStateApprox1 [HSStateApprox2 [HSStateApprox3 HSStateApprox4]]].
  auto.
Qed.

Lemma Parallel_Compositionality:
  ∀ pe l pcs envps envps' rho envs i
         (HProcessSimulation: ∀ t ac
                           (Hgetpe: PEMod.get pe t = Some ac),
             ProcessSimulation ac ac t l (pcs t) (envps t) (envps' t) rho envs i
         ),
         ProgramSimulation pe pe l pcs envps envps' rho envs i.
Proof.
  intros.
  unfold ProgramSimulation.
  intros psts psts' HPStatesApprox HPTerm1 HPTerm2 Hpepstseqdom Hpepstseqdom'.
  eapply Compositionality; eauto.
  - apply PEMod.eqdom_refl; auto.
  - intros t ac pst Hgetpe Hgetpsts.
    destruct HPStatesApprox as [HPStatesApprox1 HPStatesApprox2].
    specialize (HPStatesApprox1 t pst Hgetpsts).
    destruct HPStatesApprox1 as [pst' [Hgetpsts' HPStateApprox]].
    ∃ ac, pst'.
    split; auto.
    split; auto.
    specialize (HProcessSimulation t ac Hgetpe).
    unfold ProcessSimulation in HProcessSimulation.
    eapply HProcessSimulation; eauto.
Qed.

Lemma WellTyped_Process_ensures_ProcessSimulation:
  ∀ c rho envs pc envp envp' ac t i l
         (HTrans: Trans (c, pc, rho t, envs, envp) (ac, envp'))
         (Hl: l = gsp(rho i))
  ,
    ProcessSimulation ac ac t l pc envp envp' rho envs i.
Proof.
  intros.
  generalize dependent i.
  generalize dependent l.
  remember (
      fun (p: Cmd ×Label ×Policy×Env×Env) (p0: AnnoCmd × Env)
      ⇒ match (p, p0) with
           | ((c,pc,po,envs, envp), (ac, envp'))
             ⇒ po = rho t →
                ∀ (l : Label) (i : Tid),
                  l = gsp (rho i) → ProcessSimulation ac ac t l pc envp envp' rho envs i
         end
    ) as P.
  assert (rho t = rho t) as Hrhot by auto.
  assert (P (c,pc,rho t,envs, envp) (ac, envp') =
          ((rho t = rho t) →
           ∀ (l : Label) (i : Tid),
             l = gsp (rho i) → ProcessSimulation ac ac t l pc envp envp' rho envs i
         )) as HPx
      by (subst; auto).
  generalize dependent Hrhot.
  rewrite <- HPx.
  clear HPx.
  induction HTrans; simpl; subst; unfold ProcessSimulation in *; intros; subst.

  - inversion HTypeSafe.
    subst.
    eapply TSim_Skip; eauto.
  - eapply WellType_imply_TSim_Assign; eauto.

  - eapply WellType_imply_TSim_Get; eauto.

  - eapply WellType_imply_TSim_Put; eauto.

  - eapply WellType_imply_TSim_Out; eauto.

  - eapply TSim_Seq; eauto.
    assert (∀ sst0 : SharedState, ThreadTerm t rho ac1 pst sst0) as HTTerm1. {
      intros.
      eapply Seq_TTerm_TTerm; eauto.
    }
    assert (∀ sst0 : SharedState, ThreadTerm t rho ac1 pst' sst0) as HTTerm2. {
      intros.
      eapply Seq_TTerm_TTerm; eauto.
    }
    eauto.
  - eapply WellType_imply_TSim_If; eauto.

  - eapply WellType_imply_TSim_While; eauto.

  - eapply IHHTrans2; eauto.
    eauto with baseenv.

  - eapply WellType_imply_TSim_HSend; eauto.

  - eapply WellType_imply_TSim_LSend; eauto.

  - eapply WellType_imply_TSim_HRecv; eauto.

  - eapply WellType_imply_TSim_LRecv; eauto.
Qed.

Definition prog_pe_eqdom (p: Prog) (pe: Progexe) :=
  (∀ t c po,
      ProgMod.get p t = Some (po, c) → ∃ ac, PEMod.get pe t = Some ac
  )
  ∧
  (∀ t ac ,
      PEMod.get pe t = Some ac → ∃ c po, ProgMod.get p t = Some (po, c)
  ).


Theorem Soundness:
  ∀ p pe pcs rho envs envps
         (Heqdom: prog_pe_eqdom p pe)
         (HTransProg:
            TransProg (p, pcs, envs, envps) (rho, pe)),
    LeakFree pe rho pcs envps envs.
Proof.
  intros.
  unfold LeakFree.
  intros.
  inversion HTransProg.
  subst.
  eapply ProgramSimulation_implies_ProgramEquivalence with (envps':=envps'); eauto.
  eapply Parallel_Compositionality; eauto.
  intros.
  destruct Heqdom as [Heqdom1 Heqdom2].
  assert (∃ c po, ProgMod.get p t = Some (po, c)) as Hgetp by eauto.
  destruct Hgetp as [c [po Hgetp]].
  eapply WellTyped_Process_ensures_ProcessSimulation; eauto.
Qed.
