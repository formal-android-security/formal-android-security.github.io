Preorder

starparaSeqStep is a preorder, which means that it is reflexive and transitive.
Lemma starparaSeqStep_refl:
  ∀ t rho ac pst,
    starparaSeqStep t rho (ac, pst) (ac, pst).
Proof.
  intros.
  ∃ O.
  simpl.
  auto.
Qed.

Lemma kparaSeqStep_trans:
  ∀ k1 k2 t rho psc psc'' psc'
         (HkparaSeqStep1 : kparaSeqStep k1 t rho psc psc'')
         (HkparaSeqStep2 : kparaSeqStep k2 t rho psc'' psc'),
    kparaSeqStep (k1+k2) t rho psc psc'.
Proof.
  induction k1.
  - intros.
    simpl in ×.
    subst; auto.
  - intros.
    destruct psc as [ac pst].
    destruct psc'' as [ac'' pst''].
    destruct psc' as [ac' pst'].
    simpl in ×.
    destruct HkparaSeqStep1 as [sst [ac''' [pst''' [sst''' [HSeqStep HkparaSeqStep1]]]]].
    ∃ sst, ac''', pst''', sst'''.
    split; auto.
    eapply IHk1; eauto.
Qed.

Lemma starparaSeqStep_trans:
  ∀ t rho psc psc'' psc'
         (HstarparaSeqStep1 : starparaSeqStep t rho psc psc'')
         (HstarparaSeqStep2 : starparaSeqStep t rho psc'' psc'),
    starparaSeqStep t rho psc psc'.
Proof.
  intros.
  destruct HstarparaSeqStep1 as [k1 HkparaSeqStep1].
  destruct HstarparaSeqStep2 as [k2 HkparaSeqStep2].
  ∃ (k1+k2).
  eapply kparaSeqStep_trans; eauto.
Qed.

Relation between SeqStep and ParaSeqStep

Lemma SeqStep_kparaSeqStep:
  ∀ k t rho ac pst ac' pst' sst' ac'' pst'' sst''
         (HkparaSeqStep: kparaSeqStep k t rho (ac,pst) (ac',pst'))
         (HSeqStep: SeqStep t rho (ac', (pst', sst')) (ac'', (pst'', sst''))),
    kparaSeqStep (S k) t rho (ac,pst) (ac'',pst'').
Proof.
  induction k.
  - intros.
    simpl in ×.
    inversion HkparaSeqStep.
    subst.
    simpl in ×.
    ∃ sst', ac'', pst'', sst''.
    split; auto.
  - intros.
    simpl in ×.
    destruct HkparaSeqStep as [sst [ac''0 [pst''0 [sst''0 [HSeqStep' HkparaSeqStep]]]]].
    ∃ sst, ac''0, pst''0, sst''0.
    split; auto.
    eapply IHk; eauto.
Qed.

Lemma SeqStep_starparaSeqStep:
  ∀ t rho ac pst ac' pst' sst' ac'' pst'' sst''
         (HstarparaSeqStep: starparaSeqStep t rho (ac,pst) (ac',pst'))
         (HSeqStep: SeqStep t rho (ac', (pst', sst')) (ac'', (pst'', sst''))),
    starparaSeqStep t rho (ac,pst) (ac'',pst'').
Proof.
  intros.
  destruct HstarparaSeqStep as [k HkparaSeqStep].
  ∃ (S k).
  eapply SeqStep_kparaSeqStep; eauto.
Qed.

TPC Equal

A private state(PrivateState) is consist of two parts, the program counter process set(TPC), i.e. tpc, and a private store (PrivateStore). If a command paraSeqStep to skip, then the program counter process set tpc of the starting private state and that of the ending private state tpc' is equal, i.e. tpc = tpc'. That is
if (C, (tpc, ps)) ↪* (ASkip, (tpc', ps')), then tpc = tpc'.
This property is expressed in lemma starparaSeqStep_Skip_tpc_eq. It is an essential property when proving the simulation of if and while commands.

Lemma kparaSeqStep_Skip_State_eq:
  ∀ k pst pst' t rho ac'
         (HkparaSeqStep: kparaSeqStep k t rho (ASkip, pst) (ac', pst')),
    pst = pst'.
Proof.
  induction k.
  - intros.
    simpl in ×.
    congruence.
  - intros.
    simpl in ×.
    destruct HkparaSeqStep as [sst [ac'' [pst'' [sst'' [HSeqStep HkparaSeqStep]]]]].
    inversion HSeqStep.
    subst.
    eapply IHk; eauto.
Qed.

Lemma Seq_kparaSeqStep_Skip_k1_kparaSeqStep_k2_kparaSeqStep:
  ∀ k ac1 ac2 pst pst' t rho
    (HkparaSeqStep: kparaSeqStep k t rho (ASeq ac1 ac2, pst) (ASkip,pst')),
  ∃ pst'' k1 k2, kparaSeqStep k1 t rho (ac1,pst) (ASkip,pst'') ∧ kparaSeqStep k2 t rho (ac2,pst'') (ASkip, pst') ∧ k = k1+k2.
Proof.
  induction k.
  - intros.
    simpl in HkparaSeqStep.
    inversion HkparaSeqStep.

  - intros.
    simpl in HkparaSeqStep.
    destruct HkparaSeqStep as [sst [ac'' [pst'' [sst'' [HkparaSeqStep1 HkparaSeqStep2]]]]].
    inversion HkparaSeqStep1.
    + subst.
      assert (∃ pst''' k1' k2',
                kparaSeqStep k1' t rho (ac1',pst'') (ASkip, pst''') ∧
                kparaSeqStep k2' t rho (ac2, pst''') (ASkip, pst') ∧
                k = k1'+k2') as HkparaSeqStep_both
          by (eapply IHk; eauto).
      destruct HkparaSeqStep_both as [pst''' [k1 [k2 [HkparaSeqStep_both1 [HkparaSeqStep_both2 Hkeqsum]]]]].
      ∃ pst'''.
      ∃ (S k1), k2.
      simpl.
      split; auto.
      ∃ sst, ac1', pst'', sst''.
      split; auto.

    + subst.
      ∃ pst''.
      ∃ (S O), k.
      split; auto.
      simpl.
      ∃ sst'', ASkip, pst'', sst''.
      split; auto.
      constructor.
Qed.

Lemma Seq_kparaSeqStep_Skip:
  ∀ k ac1 ac2 pst pst' t rho
    (HkparaSeqStep: kparaSeqStep k t rho (ASeq ac1 ac2, pst) (ASkip,pst')),
  ∃ pst'', starparaSeqStep t rho (ac1,pst) (ASkip,pst'') ∧ starparaSeqStep t rho (ac2,pst'') (ASkip, pst').
Proof.
  intros.
  apply Seq_kparaSeqStep_Skip_k1_kparaSeqStep_k2_kparaSeqStep in HkparaSeqStep.
  destruct HkparaSeqStep as [pst'' [k1 [k2 [HkparaSeqStep1 [HkparaSeqStep2 Hkeqsum]]]]].
  ∃ pst''.
  split.
  - ∃ k1; auto.
  - ∃ k2; auto.
Qed.

Lemma Seq_starparaSeqStep_Skip:
  ∀ ac1 ac2 pst pst' t rho
    (HstarparaSeqStep: starparaSeqStep t rho (ASeq ac1 ac2, pst) (ASkip,pst')),
  ∃ pst'', starparaSeqStep t rho (ac1,pst) (ASkip,pst'') ∧ starparaSeqStep t rho (ac2,pst'') (ASkip, pst').
Proof.
  intros.
  destruct HstarparaSeqStep as [k HstarparaSeqStep].
  eapply Seq_kparaSeqStep_Skip; eauto.
Qed.

Lemma kparaSeqStep_Skip_tpc_eq_seq:
  ∀ k rho t ac1 ac2 ps' tpc' ps tpc
  (IHHTrans1 : ∀ (k : nat)
                (ps' : PrivateStore) (tpc' : TPC)
                (ps : PrivateStore) (tpc : TPC),
              kparaSeqStep k t rho (ac1, (tpc, ps))
                (ASkip, (tpc', ps')) → tpc = tpc')
  (IHHTrans2 : ∀ (k : nat)
                (ps' : PrivateStore) (tpc' : TPC)
                (ps : PrivateStore) (tpc : TPC),
              kparaSeqStep k t rho (ac2, (tpc, ps))
                (ASkip, (tpc', ps')) → tpc = tpc')
  (HkparaSeqStep : kparaSeqStep k t rho (ASeq ac1 ac2, (tpc, ps))
        (ASkip, (tpc', ps'))),
   tpc = tpc'.
Proof.
  intros.
      apply Seq_kparaSeqStep_Skip in HkparaSeqStep.
      destruct HkparaSeqStep as [pst'' [HkparaSeqStep1 HkparaSeqStep2]].
      destruct HkparaSeqStep1 as [k1 HkparaSeqStep1].
      destruct HkparaSeqStep2 as [k2 HkparaSeqStep2].
      destruct pst'' as [tpc'' ps''].
      assert (tpc = tpc'') by eauto.
      assert (tpc'' = tpc') by eauto.
      congruence.
Qed.

Lemma kparaSeqStep_Skip_tpc_eq_if:
  ∀ k b rho t ac1 ac2 ps' tpc' ps tpc
  (IHHTrans1 : ∀ (k : nat)
                (ps' : PrivateStore) (tpc' : TPC)
                (ps : PrivateStore) (tpc : TPC),
              kparaSeqStep k t rho (ac1, (tpc, ps))
                (ASkip, (tpc', ps')) → tpc = tpc')
  (IHHTrans2 : ∀ (k : nat)
                (ps' : PrivateStore) (tpc' : TPC)
                (ps : PrivateStore) (tpc : TPC),
              kparaSeqStep k t rho (ac2, (tpc, ps))
                (ASkip, (tpc', ps')) → tpc = tpc')
  (HkparaSeqStep: kparaSeqStep k t rho (ASeq (AIfelse b ac1 ac2) AEnd, (tpc, ps))
         (ASkip, (tpc', ps'))),
   tpc = tpc'.
Proof.
  intros.
  destruct k.
  + simpl in ×.
    inversion HkparaSeqStep.
  + simpl in ×.
    destruct HkparaSeqStep as [sst [ac'' [pst'' [sst'' [HSeqStep HkparaSeqStep]]]]].
    inversion HSeqStep; clear HSeqStep.
    subst.
    match goal with [H:SeqStep _ _ _ _ |- _ ] ⇒ rename H into HSeqStep end.
    inversion HSeqStep.
    × subst.
      assert (tpc = tpc0) by congruence.
      assert (ps = ps0) by congruence.
      subst tpc0 ps0.
      apply Seq_kparaSeqStep_Skip in HkparaSeqStep.
      destruct HkparaSeqStep as [pst'' [HkparaSeqStep1 HkparaSeqStep2]].
      destruct HkparaSeqStep1 as [k1 HkparaSeqStep1].
      destruct HkparaSeqStep2 as [k2 HkparaSeqStep2].
      destruct pst'' as [tpc'' ps''].
      eapply IHHTrans1 in HkparaSeqStep1; eauto.
      destruct k2.
      {
        simpl in ×.
        inversion HkparaSeqStep2.
      }
      {
        simpl in HkparaSeqStep2.
        destruct HkparaSeqStep2 as [sst [ac'' [pst'' [sst''0 [HSeqStep' HkparaSeqStep2]]]]].
        inversion HSeqStep'.
        subst.
        apply kparaSeqStep_Skip_State_eq in HkparaSeqStep2.
        inversion HkparaSeqStep2.
        subst.
        congruence.
      }

    
    × subst.
      assert (tpc = tpc0) by congruence.
      assert (ps = ps0) by congruence.
      subst tpc0 ps0.
      apply Seq_kparaSeqStep_Skip in HkparaSeqStep.
      destruct HkparaSeqStep as [pst'' [HkparaSeqStep1 HkparaSeqStep2]].
      destruct HkparaSeqStep1 as [k1 HkparaSeqStep1].
      destruct HkparaSeqStep2 as [k2 HkparaSeqStep2].
      destruct pst'' as [tpc'' ps''].
      eapply IHHTrans2 in HkparaSeqStep1; eauto.
      destruct k2.
      {
        simpl in ×.
        inversion HkparaSeqStep2.
      }
      {
        simpl in HkparaSeqStep2.
        destruct HkparaSeqStep2 as [sst [ac'' [pst'' [sst''0 [HSeqStep' HkparaSeqStep2]]]]].
        inversion HSeqStep'.
        subst.
        apply kparaSeqStep_Skip_State_eq in HkparaSeqStep2.
        inversion HkparaSeqStep2.
        subst.
        congruence.
      }
Qed.

Lemma kparaSeqStep_Skip_tpc_eq_Skip:
  ∀ k t rho tpc ps tpc' ps'
  (HkparaSeqStep : kparaSeqStep k t rho (ASkip, (tpc, ps))
                        (ASkip, (tpc', ps'))),

  assert ((tpc, ps) = (tpc', ps')).

Lemma kparaSeqStep_Skip_tpc_AEnd:
  ∀ k t rho ts tpc ps tpc' ps'
  (HkparaSeqStep : kparaSeqStep k t rho (AEnd, (ts::tpc, ps))
                        (ASkip, (tpc', ps'))),
  tpc' = tpc.
Proof.
  induction k; intros.
  - simpl in ×.
    inversion HkparaSeqStep.
  - simpl in ×.
    destruct HkparaSeqStep as [sst [ac'' [pst'' [sst'' [HSeqStep HkparaSeqStep]]]]].
    inversion HSeqStep.
    subst.
    assert (tpc0 = tpc')
      by (eapply kparaSeqStep_Skip_tpc_eq_Skip; eauto).
    congruence.
Qed.

Lemma kparaSeqStep_Skip_tpc_eq_while0:
  ∀ k b rho t ac ps' tpc' ps tpc
         (IHHTrans : ∀ (k : nat)
                            (ps' : PrivateStore) (tpc' : TPC)
                            (ps : PrivateStore) (tpc : TPC),
                       kparaSeqStep k t rho (ac, (tpc, ps))
                                (ASkip, (tpc', ps')) → tpc = tpc')
         (HkparaSeqStep: kparaSeqStep k t rho (AWhiledo b (ASeq ac AEnd), (tpc, ps))
                              (ASkip, (tpc', ps'))),
    tpc = tpc'.
Proof.
  induction k using nat_ind2; intros.
  - simpl in ×.
    inversion HkparaSeqStep.
  - simpl in ×.
    rename H into IHk.
    destruct HkparaSeqStep as [sst [ac'' [pst'' [sst'' [HSeqStep HkparaSeqStep]]]]].
    inversion HSeqStep.
    + subst.
      apply Seq_kparaSeqStep_Skip_k1_kparaSeqStep_k2_kparaSeqStep in HkparaSeqStep.
      destruct HkparaSeqStep as [pst'' [k1 [k2 [HkparaSeqStep1 [HkparaSeqStep2 Hkeqsum]]]]].
      apply Seq_kparaSeqStep_Skip_k1_kparaSeqStep_k2_kparaSeqStep in HkparaSeqStep1.
      destruct HkparaSeqStep1 as [pst''' [k1' [k2' [HkparaSeqStep11 [HkparaSeqStep12 Hkeqsum']]]]].
      destruct pst'' as [tpc'' ps''].
      destruct pst''' as [tpc''' ps'''].
      assert (ts::tpc0 = tpc''') by eauto.
      assert (k2 ≤ k) by omega.
      assert (tpc'' = tpc') by eauto.
      assert (tpc''' = ts::tpc) by congruence.
      subst tpc'''.
      assert (tpc'' = tpc0) by (eapply kparaSeqStep_Skip_tpc_AEnd; eauto).
      congruence.
    + subst.
      eapply kparaSeqStep_Skip_tpc_eq_Skip; eauto.
Qed.

Lemma kparaSeqStep_Skip_tpc_eq:
  ∀ k c pc rho envs envp envp' t ac tpc ps tpc' ps'
         (HTrans: Trans (c, pc, rho t, envs, envp) (ac, envp'))
         (HkparaSeqStep: kparaSeqStep k t rho (ac, (tpc, ps)) (ASkip, (tpc',ps'))),
    tpc = tpc'.
Proof.
  intros.
  generalize dependent tpc.
  generalize dependent ps.
  generalize dependent tpc'.
  generalize dependent ps'.
  generalize dependent k.
  remember (
      fun (p: Cmd ×Label ×Policy×Env×Env) (p0: AnnoCmd × Env)
      ⇒ match (p, p0) with
           | ((c,pc,po,envs, envp), (ac, envp'))
             ⇒ ∀ k ps' tpc' ps tpc,
                  kparaSeqStep k t rho (ac, (tpc, ps)) (ASkip, (tpc', ps')) →
                  tpc = tpc'
         end
    ) as P.
  assert (P (c,pc,rho t,envs, envp) (ac, envp') =
          ∀ (k: nat) (ps' : PrivateStore) (tpc' : TPC)
                  (ps : PrivateStore) (tpc : TPC) ,
            kparaSeqStep k t rho (ac, (tpc, ps)) (ASkip, (tpc', ps')) → tpc = tpc') as HPx
      by (subst; auto).
  rewrite <- HPx.
  clear HPx.

  induction HTrans; subst; intros;
    try solve [
          match goal with [H: kparaSeqStep _ _ _ _ _ |- _] ⇒ rename H into HkparaSeqStep end;
          destruct k; [
            simpl in *;
            inversion HkparaSeqStep;
            auto
          |
          simpl in HkparaSeqStep;
          destruct HkparaSeqStep as [sst [sc'' [pst'' [sst'' [HSeqStep HkparaSeqStep]]]]];
            inversion HSeqStep;
              subst;
              apply kparaSeqStep_Skip_State_eq in HkparaSeqStep;
              congruence]].

  - eapply kparaSeqStep_Skip_tpc_eq_seq with (ac1:=ac1) (ac2:=ac2); eauto.

  - eapply kparaSeqStep_Skip_tpc_eq_if with (ac1:=ac1) (ac2:=ac2); eauto.

  - eapply kparaSeqStep_Skip_tpc_eq_while0; eauto.

  - eapply IHHTrans2; eauto.
Qed.

Lemma kparaSeqStep_Skip_is_Skip:
  ∀ k pst pst' t rho ac'
         (HkparaSeqStep: kparaSeqStep k t rho (ASkip, pst) (ac', pst')),
    ac' = ASkip.
Proof.
  induction k.
  - intros.
    simpl in ×.
    congruence.
