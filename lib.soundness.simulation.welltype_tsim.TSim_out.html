Lemma WellType_imply_TSim_Out:
  ∀ e pst pst' (rho : Tid → Policy) (envs : Env) (pc : Label)
     (envp envp' : Env) (t i : Tid)
     (l : Label)
     (HTypeSafe : TypeSafe (pc, rho t, envs) envp (Out e) envp')
     (Hl: l = gsp (rho i))
  (HPStateApprox : PStateApprox pst pst' l pc envp),
  
  
  TSim (AOut e) (AOut e) pst pst' t l pc envp' rho envs i.
Proof.
  intros.
  inversion HTypeSafe.
  subst.
  apply TSimProceed.
  split.
  - intros sst sst' ac1 pst1 sst1 HSStateApprox HSeqStep HseqstII HseqstII'.
    assert (seqstateII (pst1, sst1) t rho). {
      eapply SeqStep_seqstateII; eauto.
    }
    inversion HSeqStep.
    match goal with [H: ExprInterp _ _ = _ |- _] ⇒ rename H into Hexprinterp end.
    subst sst1.
    subst.
    destruct pst' as [tpc' ps'].
    destruct sst' as [[[ss' hqs'] lqs'] alpha'].
    remember (ExprInterp e ps') as v.
    apply eq_sym in Heqv.
    destruct v as [ts' n'].
    rename Heqv into Hexprinterp'.
    ∃ ASkip,
    (tpc', ps'), (ss', hqs', lqs', (ts_add t (ts_union ts' (tpc_union tpc')), n') :: alpha').
    split.
    + ∃ (S O); simpl.
      eapply ex_intro.
      split.
      × econstructor; eauto.
      × eauto.
    + split.
      × destruct HSStateApprox as [SStoreApprox [HHMsgqueueApprox [HLMsgqueueApprox HOutStateApprox]]].
        destruct HPStateApprox as [HPStateApprox1 HPStateApprox2].
        split; auto.
        split; auto.
        split; auto.
        {
          unfold OutstateApprox in ×.
          simpl.
          remember (tid_in i (ts_add t (ts_union ts (tpc_union tpc)))) as v1.
          remember (tid_in i (ts_add t (ts_union ts' (tpc_union tpc')))) as v2.
          apply eq_sym in Heqv1.
          apply eq_sym in Heqv2.
          unfold tid_in in ×.
          unfold ts_union in Heqv1.
          unfold ts_union in Heqv2.
          destruct (Nat.eq_dec i t).
          - subst i.
            assert (v1 = true). {
              destruct v1; auto.
              apply set_mem_complete1 in Heqv1.
              destruct Heqv1.
              apply set_add_intro2; auto.
            }
            assert (v2 = true). {
              destruct v2; auto.
              apply set_mem_complete1 in Heqv2.
              destruct Heqv2.
              apply set_add_intro2; auto.
            }
            clear Heqv1; clear Heqv2.
            subst.
            assert (n = n' ∧ ts_eq ts ts') as Heqv. {
              eapply exprInterp_eq; eauto.
              match goal with [H: (_,_) = rho _ |- _] ⇒ rewrite <- H end; simpl.
              eauto with baselabel.
            }
            destruct Heqv.
            congruence.
          - unfold ts_add in Heqv1; unfold ts_add in Heqv2.
            destruct v1.
            + assert (labelleq (Lang.gsp (rho t)) (Lang.gsp (rho i))). {
                apply set_mem_correct1 in Heqv1.
                apply set_add_elim in Heqv1.
                destruct Heqv1 as [Hin1 | Hin2]; try congruence.
                apply set_union_elim in Hin2.
                destruct HseqstII as [HpstII HsstII].
                destruct HpstII as [HtpcII HpsII].
                destruct Hin2 as [Hin21 | Hin22].
                - eapply exprInterp_in_var_in in Hin21; eauto.
                  destruct Hin21 as [x [ts'' [n'' [Hinx [Hpsx Hin]]]]].
                  eauto.
                - eapply in_tpc_union_in_tpc in Hin22.
                  destruct Hin22 as [ts'' [Hints Hin]].
                  eauto.
              }
              assert (labelleq l0 (Lang.gsp (rho i))). {
                assert (labelleq l0 (Lang.gsp (rho t))). {
                  match goal with [H: (_,_) = rho _ |- _] ⇒ rewrite <- H in *; simpl in × end.
                  eauto with baselabel.
                }
                eapply labelleq_trans with (l2:=(Lang.gsp (rho t))); auto.
              }
              assert (n = n' ∧ ts_eq ts ts') as Heqv. {
                eapply exprInterp_eq; eauto.
              }
              destruct Heqv.
              unfold ts_eq in ×.
              subst.
              assert (tpc = tpc'). {
                eapply HPStateApprox1; auto.
                match goal with [H: (_,_) = rho _ |- _] ⇒ rewrite <- H in × end; simpl in ×.
                eauto with baselabel.
              }
              subst.
              rewrite Heqv1.
              congruence.
            + destruct v2.
              × assert (labelleq (Lang.gsp (rho t)) (Lang.gsp (rho i))). {
                  apply set_mem_correct1 in Heqv2.
                  apply set_add_elim in Heqv2.
                  destruct Heqv2 as [Hin1 | Hin2]; try congruence.
                  apply set_union_elim in Hin2.
                  destruct HseqstII' as [HpstII' HsstII'].
                  destruct HpstII' as [HtpcII' HpsII'].
                  destruct Hin2 as [Hin21 | Hin22].
                  - eapply exprInterp_in_var_in in Hin21; eauto.
                    destruct Hin21 as [x [ts'' [n'' [Hinx [Hpsx Hin]]]]].
                    eauto.
                  - eapply in_tpc_union_in_tpc in Hin22.
                    destruct Hin22 as [ts'' [Hints Hin]].
                    eauto.
                }
                assert (labelleq l0 (Lang.gsp (rho i))). {
                  assert (labelleq l0 (Lang.gsp (rho t))). {
                    match goal with [H: (_,_) = rho _ |- _] ⇒ rewrite <- H in *; simpl in × end.
                    eauto with baselabel.
                  }
                  eapply labelleq_trans with (l2:=(Lang.gsp (rho t))); auto.
                }
                assert (n = n' ∧ ts_eq ts ts') as Heqv. {
                  eapply exprInterp_eq; eauto.
                }
                destruct Heqv.
                unfold ts_eq in ×.
                subst.
                assert (tpc = tpc'). {
                  eapply HPStateApprox1; auto.
                  match goal with [H: (_,_) = rho _ |- _] ⇒ rewrite <- H in × end; simpl in ×.
                  eauto with baselabel.
                }
                subst.
                congruence.
              × auto.
        }
      × split; auto.
        eapply TSim_Skip; auto.
  - intros.
    congruence.
Qed.
