The State Equivalence

Private State Equivalence

Definition PStoreApprox (ps1 ps2 : PrivateStore) (l:Label) (envp: Env):=
  ∀ x ts1 n1 ts2 n2 lx
         (Hlx: lx = envp x)
         (Hps1x: ps1(x) = (ts1, n1))
         (Hps2x: ps2(x) = (ts2, n2))
         (Hlabelleqlxl: labelleq lx l),
    (n1=n2 ∧ ts_eq ts1 ts2).

Definition PStateApprox (pst1 pst2 : PrivateState) (l:Label) (pc: Label)(envp:Env) :=
  match (pst1, pst2) with
  | ((tpc1, ps1), (tpc2, ps2)) ⇒
    (labelleq pc l → tpc1 = tpc2) ∧
    (PStoreApprox ps1 ps2 l envp)
  end.

Definition PStatesApprox (psts1 psts2 : PrivateStateSet) (l:Label) (pcs: PC_Map)(envps: Env_Map) :=
  (∀ t pst1,
      PSTSMod.get psts1 t = Some pst1 →
      ∃ pst2, PSTSMod.get psts2 t = Some pst2 ∧
                   PStateApprox pst1 pst2 l (pcs t)(envps t)
  ) ∧
  (∀ t pst2,
      PSTSMod.get psts2 t = Some pst2 →
      ∃ pst1, PSTSMod.get psts1 t = Some pst1 ∧
                   PStateApprox pst1 pst2 l (pcs t)(envps t)
  )
.

Shared State Equivalence


Definition SStoreApprox (ss1 ss2 : SharedStore) (l : Label) (envs: Env):=
  ∀ x ts1 n1 lx ts2 n2
         (Hlx: lx = envs x)
         (Hss1x: ss1(x) = (ts1, n1))
         (Hss2x: ss2(x) = (ts2, n2))
         (Hlabelleqlxl: labelleq lx l),
    (n1=n2 ∧ ts_eq ts1 ts2).

Definition HMsgqueueApprox (hq1 hq2 : HMsgQueue) (l:Label) (lm: Label) :=
  if (labelleq_dec lm l)
  then hq1 = hq2
  else True.

Definition HMsgqueueSetApprox (hqs1 hqs2 : HMsgQueueSet) (l:Label) (rho: Policys) :=
  (∀ t hq1,
      HMSGSMod.get hqs1 t = Some hq1 →
      ∃ hq2, HMSGSMod.get hqs2 t = Some hq2 ∧
                  HMsgqueueApprox hq1 hq2 l (gap(rho(t)))
  )
  ∧
  (∀ t hq2,
      HMSGSMod.get hqs2 t = Some hq2 →
      ∃ hq1, HMSGSMod.get hqs1 t = Some hq1 ∧
                  HMsgqueueApprox hq1 hq2 l (gap(rho(t)))
  )
.

Definition LMsgqueueApprox (lq1 lq2 : LMsgQueue) :=
  lq1 = lq2.

Definition LMsgqueueSetApprox (lqs1 lqs2 : LMsgQueueSet) :=
  (∀ t lq1,
      LMSGSMod.get lqs1 t = Some lq1 →
      ∃ lq2, LMSGSMod.get lqs2 t = Some lq2 ∧
                  LMsgqueueApprox lq1 lq2
  )
  ∧
  (∀ t lq2,
      LMSGSMod.get lqs2 t = Some lq2 →
      ∃ lq1, LMSGSMod.get lqs1 t = Some lq1 ∧
                  LMsgqueueApprox lq1 lq2
  )
.

Fixpoint suboutstate (i:Tid) (a:OutState) :=
  match a with
  | nil ⇒ nil
  | (ts,n)::a' ⇒ if (tid_in i ts)
                  then n::(suboutstate i a')
                  else suboutstate i a'
  end.

Definition OutstateApprox (a1 a2 :OutState) (i:Tid) :=
  suboutstate i a1 = suboutstate i a2.

Definition SharedStateApprox (sst1 sst2 : SharedState) (i:Tid) (l:Label) (rho:Policys) (envs: Env) :=
  match (sst1, sst2) with
  | ((ss1,hqs1,lqs1,alpha1), (ss2,hqs2,lqs2,alpha2)) ⇒
    SStoreApprox ss1 ss2 l envs ∧
    HMsgqueueSetApprox hqs1 hqs2 l rho ∧
    LMsgqueueSetApprox lqs1 lqs2 ∧
    OutstateApprox alpha1 alpha2 i
  end.

The Whole State Equivalence

Definition StateApprox (sigma1 sigma2 : State) (i:Tid) (l:Label) (rho: Policys) (pcs: PC_Map) (envps:Env_Map) (envs: Env) :=
  match (sigma1, sigma2) with
  | ((psts1, sst1), (psts2, sst2)) ⇒
    PStatesApprox psts1 psts2 l pcs envps ∧
    SharedStateApprox sst1 sst2 i l rho envs
  end.

Termination


Inductive ThreadTerm: Tid → Policys → AnnoCmd → PrivateState → SharedState → Prop :=
| Skip_TTerm:
    ∀ t rho pst sst, ThreadTerm t rho ASkip pst sst
| Ind_TTerm:
    ∀ t rho ac pst sst ac' pst' sst',
      SeqStep t rho (ac, (pst, sst)) (ac', (pst', sst')) →
      (∀ sst'', ThreadTerm t rho ac' pst' sst'') →
      ThreadTerm t rho ac pst sst.

Definition ProgTerm (rho: Policys) (pe: Progexe) (psts: PrivateStateSet) :=
  ∀ t ac pst
         (Hgetpsts: PSTSMod.get psts t = Some pst)
         (Hgetpe: PEMod.get pe t = Some ac),
  ∀ sst, ThreadTerm t rho ac pst sst.

Invariant

Definition pstoreII (ps: PrivateStore) (t: Tid) (rho: Policys) :=
  ∀ x ts n i,
    ps(x) = (ts, n) →
    In i ts →
    i ≠ t →
    labelleq (gsp (rho t)) (gsp (rho i)).

Definition tpcII (tpc: TPC) (t: Tid) (rho: Policys) :=
  ∀ i curtpc,
    i ≠ t →
    In curtpc tpc →
    In i curtpc →
    labelleq (gsp (rho t)) (gsp (rho i)).

Definition pstateII (pst: PrivateState) (t: Tid) (rho: Policys) :=
  match pst with
  | (tpc, ps) ⇒ tpcII tpc t rho ∧ pstoreII ps t rho
  end.

Definition pstatesII (psts: PrivateStateSet) (rho: Policys) :=
  ∀ t pst,
    PSTSMod.get psts t = Some pst →
    pstateII pst t rho .

Definition hmsgqII (hmsgq: HMsgQueue) (t: Tid) (rho: Policys) :=
  ∀ i msg,
    In msg hmsgq →
    (match msg with
     | (i', ts, _) ⇒ i' = i ∨ In i ts
     end) →
    i ≠ t →
    labelleq (gsp (rho t)) (gsp (rho i)).

Definition lmsgqII (lmsgq: LMsgQueue) (t: Tid) (rho: Policys) :=
  ∀ i msg,
    In msg lmsgq →
    (match msg with
     | (i', ts, _) ⇒ i' = i ∨ In i ts
     end) →
    i ≠ t →
    labelleq (gsp (rho t)) (gsp (rho i)).

Definition hmsgqsII(hmsgqs: HMsgQueueSet) (rho: Policys) :=
  ∀ t hmsgq,
    HMSGSMod.get hmsgqs t = Some hmsgq →
    hmsgqII hmsgq t rho.

Definition lmsgqsII(lmsgqs: LMsgQueueSet) (rho: Policys) :=
  ∀ t lmsgq,
    LMSGSMod.get lmsgqs t = Some lmsgq →
    lmsgqII lmsgq t rho.

Definition sstateII (sst: SharedState) (rho: Policys) :=
  match sst with
  | (_, hqs, lqs, _) ⇒ hmsgqsII hqs rho ∧ lmsgqsII lqs rho
  end.

Definition seqstateII (seqstate: PrivateState×SharedState) (t: Tid) (rho: Policys) :=
  match seqstate with
  | (pst, sst) ⇒ pstateII pst t rho ∧ sstateII sst rho
  end.

Definition stateII (sigma: State) (rho: Policys) :=
  match sigma with
  | (psts, sst) ⇒ pstatesII psts rho ∧ sstateII sst rho
  end.

Program Equivalence

Definition peSkip (pe: Progexe) :Progexe :=
  PEMod.mapmap pe (fun ac ⇒ ASkip).

Definition ProgEqu (pe pe': Progexe) (i:Tid) (l:Label) (rho: Policys) (pcs: PC_Map) (envps:Env_Map) (envs: Env) :=
  ∀ sigma psts sst sigma' psts' sst' psts1 ss1 hqs1 lqs1 alpha1 ,
    sigma = (psts, sst) →
    sigma' = (psts', sst') →
    ProgTerm rho pe psts →
    ProgTerm rho pe' psts' →
    stateII (psts,sst) rho →
    stateII (psts',sst') rho →
    StateApprox sigma sigma' i l rho pcs envps envs →
    starParaStep rho (pe, sigma) (peSkip pe, (psts1, (ss1, hqs1, lqs1, alpha1))) →
    PEPSTSMod.eqdom pe psts →
    PEPSTSMod.eqdom pe' psts' →
    (
      ∃ psts2 ss2 hqs2 lqs2 alpha2,
        starParaStep rho (pe', sigma') (peSkip pe', (psts2, (ss2, hqs2, lqs2, alpha2)))
        ∧
        OutstateApprox alpha1 alpha2 i
    ).

Leakage Freedom

Leakage freedom is a non-inteference-like property, which states that two executions starting from the equivalent states outputs the same informations.
Definition LeakFree (pe: Progexe) (rho: Policys) (pcs: PC_Map) (envps:Env_Map) (envs: Env) :=
  ∀ i, ProgEqu pe pe i (gsp (rho i)) rho pcs envps envs.
