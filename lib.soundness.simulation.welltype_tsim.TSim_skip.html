Require Import Simulation.

Lemma TSim_Skip:
  ∀ t l pc envp rho envs i pst pst'
         (HPStateApprox: PStateApprox pst pst' l pc envp),
    TSim ASkip ASkip pst pst' t l pc envp rho envs i.
Proof.
  cofix Hc.
  intros.
  apply TSimProceed.
  split.
  intros sst sst' ac1 pst1 sst1 HSStateApprox HSeqStep.
  inversion HSeqStep.
  subst.
  ∃ ASkip, pst', sst'.
  split; auto.
  ∃ O; simpl; auto.

  intros.
  ∃ pst', sst'.
  split.
  ∃ O; simpl; auto.
  split; auto with basestate.
Qed.
