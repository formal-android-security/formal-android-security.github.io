Lemma WellType_imply_TSim_LRecv_labelleq_pc_l:
  ∀ x y pst pst' (rho : Tid → Policy) (envs : Env) (pc : Label)
         (envp envp' : Env) (t i : Tid)
         (l : Label)
         (Hlabelleq: labelleq pc l)
         (HTypeSafe : TypeSafe (pc, rho t, envs) envp (Lrecv x y) envp')
         (HPStateApprox : PStateApprox pst pst' l pc envp),
         
         
    TSim (ALrecv x y) (ALrecv x y) pst pst' t l pc envp' rho envs i.
Proof.
  intros.
  inversion HTypeSafe.
  subst.
  apply TSimProceed.
  split.
  - intros sst sst' ac1 pst1 sst1 HSStateApprox HSeqStep HseqstII HseqstII'.
    assert (seqstateII (pst1, sst1) t rho). {
      eapply SeqStep_seqstateII; eauto.
    }
    inversion HSeqStep.
    + subst sst1.
      subst.
      destruct pst' as [tpc' ps'].
      destruct HPStateApprox as [HPCApprox HPStoreApprox].
      destruct sst' as [[[ss' hqs'] lqs'] alpha'].
      destruct HSStateApprox as [HSStoreApprox [HHMsgqueuesApprox [HLMsgqueuesApprox HOutStateApprox]]].
      destruct HLMsgqueuesApprox as [HLMsgqueuesApprox1 HLMsgqueuesApprox2].
      assert (∃ lqs't, LMSGSMod.get lqs' t = Some lqs't ∧ LMsgqueueApprox ((i0,ts,n)::lqt') lqs't) as Hgetlqs't. {
        eapply HLMsgqueuesApprox1; eauto.
      }
      destruct Hgetlqs't as [lqs't [Hgetlqs't HLMsgqueueApprox]].
      assert (tpc = tpc') by eauto.
      unfold LeakFree.LMsgqueueApprox in HLMsgqueueApprox.
       subst.
       assert (SeqStep t rho (ALrecv x y, (tpc', ps', (ss', hqs', lqs', alpha'))) (ASkip, (tpc', updateps (updateps ps' x (ts_add t (tpc_union tpc'), i0)) y (ts_add t (ts_union ts (tpc_union tpc')), n), (ss', hqs', LMSGSMod.set lqs' t lqt', alpha')))). {
         econstructor; eauto.
       }
       assert (seqstateII (tpc',
         updateps (updateps ps' x (ts_add t (tpc_union tpc'), i0)) y
           (ts_add t (ts_union ts (tpc_union tpc')), n),
         (ss', hqs', LMSGSMod.set lqs' t lqt', alpha')) t rho). {
         eapply SeqStep_seqstateII; eauto.
       }
         
        ∃ ASkip,
        (tpc', updateps (updateps ps' x (ts_add t (tpc_union tpc'), i0)) y (ts_add t (ts_union ts (tpc_union tpc')), n)),
        (ss', hqs',LMSGSMod.set lqs' t lqt', alpha').
        split.
        -- ∃ (S O).
           simpl.
           eapply ex_intro.
           split; eauto.
        -- split.
           ++ split; auto.
              split; auto.
              split; auto.
              split.
              ** intros.
                 destruct (LMSGSMod.eq_dec t t0).
                 --- subst.
                     ∃ lq1.
                     rewrite LMSGSMod.set_a_get_a in *; auto with basemap.
                 --- rewrite LMSGSMod.set_a_get_a' in *; auto with basemap.
              ** intros.
                 destruct (LMSGSMod.eq_dec t t0).
                 --- subst.
                     ∃ lq2.
                     rewrite LMSGSMod.set_a_get_a in *; auto with basemap.
                 --- rewrite LMSGSMod.set_a_get_a' in *; auto with basemap.
           ++ split; auto.
              eapply TSim_Skip; eauto.
              ** split; auto.
                 unfold PStoreApprox.
                 intros.
                 unfold updateps in ×.
                 unfold updateenv in ×.
                 destruct (Nat.eq_dec y x0); destruct (Nat.eq_dec x x0); try solve [split; congruence].
                 assert (lx = envp x0). {
                   congruence.
                 }
                 eapply HPStoreApprox; eauto.
    + subst sst1.
      subst.
      destruct pst' as [tpc' ps'].
      destruct HPStateApprox as [HPCApprox HPStoreApprox].
      destruct sst' as [[[ss' hqs'] lqs'] alpha'].
      destruct HSStateApprox as [HSStoreApprox [HHMsgqueuesApprox [HLMsgqueuesApprox HOutStateApprox]]].
      destruct HLMsgqueuesApprox as [HLMsgqueuesApprox1 HLMsgqueuesApprox2].
      assert (∃ lqs't, LMSGSMod.get lqs' t = Some lqs't ∧ LMsgqueueApprox nil lqs't) as Hgetlqs't. {
        eapply HLMsgqueuesApprox1; eauto.
      }
      destruct Hgetlqs't as [lqs't [Hgetlqs't HLMsgqueueApprox]].
      assert (tpc = tpc') by eauto.
      unfold LeakFree.LMsgqueueApprox in HLMsgqueueApprox.
      subst.
      assert (SeqStep t rho (ALrecv x y, (tpc', ps', (ss', hqs', lqs', alpha'))) (ASkip, (tpc', updateps (updateps ps' x (ts_add t (tpc_union tpc'), 0)) y (ts_add t (tpc_union tpc'), 0), (ss', hqs', lqs', alpha')))). {
        eapply SSLrecvfail; eauto.
      }
      assert (seqstateII
         (tpc',
         updateps (updateps ps' x (ts_add t (tpc_union tpc'), 0)) y
                  (ts_add t (tpc_union tpc'), 0), (ss', hqs', lqs', alpha')) t rho). {
        eapply SeqStep_seqstateII; eauto.
      }
        
      ∃ ASkip,
        (tpc', updateps (updateps ps' x (ts_add t (tpc_union tpc'), 0)) y (ts_add t (tpc_union tpc'), 0)),
        (ss', hqs', lqs', alpha').
        split.
        -- ∃ (S O).
           simpl.
           eapply ex_intro.
           split; eauto.
        -- split.
           ++ split; auto.
           ++ split; auto.
             eapply TSim_Skip; eauto.
              ** split; auto.
                 unfold PStoreApprox.
                 intros.
                 unfold updateps in ×.
                 unfold updateenv in ×.
                 destruct (Nat.eq_dec y x0); destruct (Nat.eq_dec x x0); try solve [split; congruence].
                 assert (lx = envp x0). {
                   congruence.
                 }
                 eapply HPStoreApprox; eauto.

  - intros.
    congruence.
Qed.

Lemma WellType_imply_TSim_LRecv:
  ∀ x y pst pst' (rho : Tid → Policy) (envs : Env) (pc : Label)
         (envp envp' : Env) (t i : Tid)
         (l : Label)
         (HTypeSafe : TypeSafe (pc, rho t, envs) envp (Lrecv x y) envp')
         (HPStateApprox : PStateApprox pst pst' l pc envp),
         
         
    TSim (ALrecv x y) (ALrecv x y) pst pst' t l pc envp' rho envs i.
Proof.
  intros.
  assert (labelleq pc l). {
    inversion HTypeSafe.
    subst.
    eauto with baselabel.
  }
  eapply WellType_imply_TSim_LRecv_labelleq_pc_l; eauto.
Qed.
