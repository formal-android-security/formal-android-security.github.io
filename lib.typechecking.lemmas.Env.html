Definition eq_env (env1 env2 : Env) := ∀ x, labeleq (env1 x) (env2 x).

Hint Unfold leq_env: baseenv.
Hint Unfold eq_env: baseenv.
Hint Unfold env_union: baseenv.

Lemma leq_env_refl:
  ∀ envp, leq_env envp envp.
Proof.
  auto with baselabel baseenv.
Qed.

Hint Resolve leq_env_refl: baseenv.

Lemma nlabelleq_env_union_l:
  ∀ envp1 envp2 l x
         (Hnlabelleq: ¬ labelleq l (env_union envp1 envp2 x)),
    ¬ labelleq l (envp1 x).
Proof.
  intros.
  intro Hnlabelleq'.
  destruct Hnlabelleq.
  assert (labelleq (envp1 x) (env_union envp1 envp2 x))
    by eauto with baseenv.
  eauto with baselabel.
Qed.

Lemma nlabelleq_env_union_r:
  ∀ envp1 envp2 l x
         (Hnlabelleq: ¬ labelleq l (env_union envp1 envp2 x)),
    ¬ labelleq l (envp2 x).
Proof.
  intros.
  intro Hnlabelleq'.
  destruct Hnlabelleq.
  assert (labelleq (envp2 x) (env_union envp1 envp2 x))
    by eauto with baseenv.
  eauto with baselabel.
Qed.

Hint Resolve nlabelleq_env_union_l: baseenv.
Hint Resolve nlabelleq_env_union_r: baseenv.


Lemma env_union_x_eq:
  ∀ envp1 envp2 envp x
         (Henvpxeq1: envp1 x = envp x)
         (Henvpxeq2: envp2 x = envp x),
    env_union envp1 envp2 x = envp x.
Proof.
  intros.
  unfold env_union.
  rewrite Henvpxeq1.
  rewrite Henvpxeq2.
  eapply lunion_l_l_eq_l; eauto.
Qed.

Hint Resolve env_union_x_eq: baseenv.

Qed.
