Lemma kSeqStep_imply_kParaStep :
  ∀ k i pe rho psts pst sst pst' sst' ac ac',
    PEMod.get pe i = Some ac →
    PSTSMod.get psts i = Some pst →
    kSeqStep k i rho (ac, (pst, sst)) (ac', (pst', sst')) →
    kParaStep k rho (pe, (psts, sst)) (updatepe pe i ac', ((updatepsts psts i pst'), sst')).
Proof.
  induction k.

  crush.
  match goal with [H: (_, (_,_)) = (_, (_, _)) |- _] ⇒ inversion H end.
  subst.
  assert (updatepe pe i ac' = pe) by auto with basemap.
  assert (updatepsts psts i pst' = psts) by auto with basemap.
  crush.

  intros.
  crush.
  destruct x as [ac'' [pst'' sst'']].
  ∃ (updatepe pe i ac'', (updatepsts psts i pst'', sst'')).
  split.
  eapply PStep; eauto.
  assert (updatepe (updatepe pe i ac'') i ac' = updatepe pe i ac') as Heq1 by auto with basemap.
  assert (updatepsts (updatepsts psts i pst'') i pst' = updatepsts psts i pst') as Heq2 by auto with basemap.
  rewrite <- Heq1 .
  rewrite <- Heq2 .
  eapply IHk with (pe:=updatepe pe i ac'') (psts:=updatepsts psts i pst''); eauto with basemap.
Qed.


Definition starSeqStep_imply_starParaStep:
  ∀ i pe rho psts pst sst pst' sst' ac ac',
    PEMod.get pe i = Some ac →
    PSTSMod.get psts i = Some pst →
    starSeqStep i rho (ac, (pst, sst)) (ac', (pst', sst')) →
    starParaStep rho (pe, (psts, sst)) (updatepe pe i ac', ((updatepsts psts i pst'), sst')).
Proof.
  intros.
  match goal with [H: starSeqStep _ _ _ _ |- _] ⇒ rename H into HstarSeqStep end.
  destruct HstarSeqStep as [k HkSeqStep].
  ∃ k.
  eapply kSeqStep_imply_kParaStep; eauto.
Qed.

Lemma ParaStep_merge_ParaStep_l:
  ∀ pe1 pe2 psts1 psts2 sst pe1' psts1' sst' rho
    (HParaStep: ParaStep rho (pe1, (psts1, sst)) (pe1', (psts1', sst'))),
       ParaStep rho (PEMod.merge pe1 pe2, (PSTSMod.merge psts1 psts2, sst)) (PEMod.merge pe1' pe2, (PSTSMod.merge psts1' psts2, sst')).
Proof.
  intros.
  inversion HParaStep.
  subst.
  rename H6 into Hgetpsts.
  rename H7 into Hgetpe.
  eapply PStep; eauto.

  rewrite PSTSMod.merge_sem.
  rewrite Hgetpsts.
  destruct (PSTSMod.get psts2 i); auto.

  rewrite PEMod.merge_sem.
  rewrite Hgetpe.
  destruct (PEMod.get pe2 i); auto.

  unfold updatepe.
  apply PEMod.merge_set_eq_1.

  unfold updatepsts.
  apply PSTSMod.merge_set_eq_1.
Qed.

Lemma ParaStep_merge_ParaStep_r:

    (HParaStep: ParaStep rho (pe2, (psts2, sst)) (pe2', (psts2', sst')))
         (Hpedisj: PEMod.disj pe1 pe2)
         (Hpstsdisj: PSTSMod.disj psts1 psts2),
       ParaStep rho (PEMod.merge pe1 pe2, (PSTSMod.merge psts1 psts2, sst)) (PEMod.merge pe1 pe2', (PSTSMod.merge psts1 psts2', sst')).

  subst.
  rename H6 into Hgetpsts.
  rename H7 into Hgetpe.

  rewrite PSTSMod.merge_sem.
  rewrite Hgetpsts.
  unfold PSTSMod.disj in Hpstsdisj.
  specialize (Hpstsdisj i).
  rewrite Hgetpsts in Hpstsdisj.
  destruct (PSTSMod.get psts1 i); [exfalso; auto | auto].

  rewrite PEMod.merge_sem.
  rewrite Hgetpe.
  unfold PEMod.disj in Hpedisj.
  specialize (Hpedisj i).
  rewrite Hgetpe in Hpedisj.

Lemma SeqStep_determine:
  ∀ t rho ac seqstate ac' seqstate' ac'' seqstate''
    (HSeqStep1: SeqStep t rho (ac, seqstate) (ac', seqstate'))
    (HSeqStep2: SeqStep t rho (ac, seqstate) (ac'', seqstate''))
    (Hacneq1: ac ≠ ac')
    (Hacneq2: ac ≠ ac''),
    ac' = ac'' ∧ seqstate' = seqstate''.
Proof.
  intros.
  generalize dependent ac''.
  generalize dependent seqstate''.
  generalize dependent Hacneq1.

  remember (
      fun t rho sc sc' ⇒
        match (sc, sc') with
        | ((ac, seqstate), (ac', seqstate')) ⇒
          ac ≠ ac' →
          ∀ (seqstate'' : PrivateState × SharedState) (ac'' : AnnoCmd),
            SeqStep t rho (ac, seqstate) (ac'', seqstate'') →
            ac ≠ ac'' → ac' = ac'' ∧ seqstate' = seqstate''
        end
    )
    as P.
  assert (P t rho (ac,seqstate) (ac', seqstate') =
          (ac ≠ ac' →
          ∀ (seqstate'' : PrivateState × SharedState) (ac'' : AnnoCmd),
            SeqStep t rho (ac, seqstate) (ac'', seqstate'') →
            ac ≠ ac'' → ac' = ac'' ∧ seqstate' = seqstate'')
         ) as HP
               by (subst;simpl; auto).
  rewrite <- HP; clear HP.

  induction HSeqStep1;
    try solve [
          subst; simpl in *; intros;
          match goal with [H: SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep2 end;
          inversion HSeqStep2;
          split; congruence].

  - subst; simpl; intros.
    match goal with [H: SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep2 end.
    inversion HSeqStep2; clear HSeqStep2.
    + subst.
      assert (ac1' = ac1'0 ∧ seqstate'0 = seqstate'') as Heqv.
      {
        eapply IHHSeqStep1; congruence.
      }
      destruct Heqv.
      split; congruence.
    + subst.
      inversion HSeqStep1.
      subst.
      congruence.
  - subst; simpl; intros.
    match goal with [H: SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep2 end.
    inversion HSeqStep2; clear HSeqStep2.
    + subst.
      match goal with [H: SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep end.
      inversion HSeqStep.
      subst.
      congruence.
    + auto.
Qed.

Lemma SeqStep_determine_ac:
  ∀ t rho ac seqstate ac' seqstate' ac'' seqstate''
    (HSeqStep1: SeqStep t rho (ac, seqstate) (ac', seqstate'))
    (HSeqStep2: SeqStep t rho (ac, seqstate) (ac'', seqstate''))
    (Hacneq1: ac ≠ ac')
    (Hacneq2: ac ≠ ac''),
    ac' = ac''.
Proof.
  intros.
  assert (ac' = ac'' ∧ seqstate' = seqstate'') as Heqv
    by (eapply SeqStep_determine; eauto).
  destruct Heqv; auto.
Qed.

Lemma SeqStep_ac_eq_seqstate_eq:
  ∀ ac seqstate ac' seqstate' t rho
         (Haceq: ac = ac')
         (HSeqStep: SeqStep t rho (ac, seqstate) (ac', seqstate')),
    seqstate = seqstate'.
Proof.
  intros.
  generalize dependent Haceq.
  remember (
      fun (t:Tid) (rho:Policys) (sc sc': SeqConfig) ⇒
        match (sc, sc') with
        | ((ac, seqstate), (ac', seqstate')) ⇒
            ac = ac' → seqstate = seqstate'
        end
    )
    as P.
  assert (P t rho (ac,seqstate) (ac', seqstate') =
          (ac = ac' → seqstate = seqstate')
         ) as HP
      by (subst; simpl; auto).
  rewrite <- HP; clear HP.
  induction HSeqStep; subst; intros; try congruence.

  - apply IHHSeqStep.
    congruence.
  - exfalso.
    eapply If_neq1; eauto.
  - exfalso.
    eapply If_neq2; eauto.
Qed.

Lemma Seq_kSeqStep_Skip_k1_kSeqStep_k2_kSeqStep:
  ∀ k ac1 ac2 seqstate seqstate' t rho
    (HkSeqStep: kSeqStep k t rho (ASeq ac1 ac2, seqstate) (ASkip,seqstate')),
  ∃ seqstate'' k1 k2, kSeqStep k1 t rho (ac1,seqstate) (ASkip,seqstate'') ∧ kSeqStep k2 t rho (ac2,seqstate'') (ASkip, seqstate') ∧ k = k1+k2.
Proof.
  induction k.
  - intros.
    simpl in HkSeqStep.
    inversion HkSeqStep.

  - intros.
    simpl in HkSeqStep.
    destruct HkSeqStep as [sc [HkSeqStep1 HkSeqStep2]].
    inversion HkSeqStep1.
ASeq ac1 ac2
    + subst.
      assert (∃ seqstate'' k1' k2',
                kSeqStep k1' t rho (ac1',seqstate'0) (ASkip, seqstate'') ∧
                kSeqStep k2' t rho (ac2, seqstate'') (ASkip, seqstate') ∧
                k = k1'+k2') as HkSeqStep_both
          by (eapply IHk; eauto).
      destruct HkSeqStep_both as [seqstate'' [k1 [k2 [HkSeqStep_both1 [HkSeqStep_both2 Hkeqsum]]]]].
      ∃ seqstate''.
      ∃ (S k1), k2.
      simpl.
      split; auto.
      ∃ (ac1', seqstate'0).
      split; auto.

ASeq ASkip ac2
    + subst.
      ∃ seqstate.
      ∃ (S O), k.
      split; auto.
      simpl.
      ∃ (ASkip, seqstate).
      split; auto.
      constructor.
Qed.
