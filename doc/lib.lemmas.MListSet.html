Lemma set_add_in_eq:
  ∀ (A:Type) (a: A) (l: list A)
         (HAdec: ∀ (x y: A), {x=y}+{x≠y})
         (Hin: In a l),
    set_add HAdec a l = l.
Proof.
  intros.
  induction l.
  - inversion Hin.
  - simpl in ×.
    destruct Hin as [Heq | Hin].
    + subst.
      destruct (HAdec a a).
      × auto.
      × congruence.
    + destruct (HAdec a a0).
      × auto.
      × rewrite IHl; auto.
Qed.

Hint Resolve set_add_in_eq: baseset.

Lemma set_union_nil_eq:
  ∀ (A:Type) (l: list A)
         (HAdec: ∀ (x y: A), {x=y}+{x≠y}),
    set_union HAdec l nil = l.
Proof.
  induction l.
  - intros.
    auto.
  - intros.
    simpl.
    auto.
Qed.

Hint Resolve set_union_nil_eq: baseset.

Lemma incl_nil_eq_nil:
  ∀ A (l:list A)
         (Hincl: incl l nil),
    l = nil.
Proof.
  intros.
  destruct l.
  - auto.
  - unfold incl in ×.
    specialize (Hincl a).
    specialize (Hincl (in_eq a l)).
    inversion Hincl.
Qed.

Hint Resolve incl_nil_eq_nil: baseset.

Lemma incl_cons_in:
  ∀ (A: Type) (a: A) (l1 l2: list A)
    (Hincl : incl (a :: l1) l2),
    In a l2.
Proof.
  intros.
  unfold incl in ×.
  specialize (Hincl a).
  specialize (Hincl (in_eq a l1)).
  auto.
Qed.

Hint Resolve incl_cons_in: baseset.

Lemma incl_cons_incl:
  ∀ (A: Type) (a: A) (l1 l2: list A)
    (Hincl : incl (a :: l1) l2),
    incl l1 l2.
Proof.
  unfold incl in ×.
  intros.
  specialize (Hincl a0).
  apply Hincl.
  eapply in_cons; eauto.
Qed.

Hint Resolve incl_cons_incl: baseset.

Lemma incl_union_eq:
  ∀ (A:Type) (l1 l2: list A)
         (HAdec: ∀ (x y: A), {x=y}+{x≠y})
         (Hincl: incl l1 l2),
    set_union HAdec l2 l1 = l2.
Proof.
  induction l1.
  - intros.
    simpl.
    auto.
  - intros.
    simpl in ×.
    assert (incl l1 l2). {
      eapply incl_cons_incl; eauto.
    }
    rewrite IHl1; auto.
    assert (In a l2). {
      eapply incl_cons_in; eauto.
    }
    rewrite set_add_in_eq; auto.
Qed.

Hint Resolve incl_union_eq: baseset.

Lemma set_union_s_s_eq_s:
  ∀ (A:Type) s
         (HAdec: ∀ (x y: A), {x=y}+{x≠y}),
    set_union HAdec s s = s.
Proof.
  intros.
  eapply incl_union_eq; auto.
  eapply incl_refl.
Qed.

Hint Resolve set_union_s_s_eq_s: baseset.

Lemma incl_dec:
  ∀ (A:Type)
         (HA_dec: ∀ x y:A, {x=y} + {x≠y})
         (l l': list A),
    {incl l l'} + {¬incl l l'}.
Proof.
  induction l.

  intros.
  left.
  unfold incl.
  intros.
  inversion H.

  intros.
  specialize (IHl l').
  destruct IHl; auto.
incl l l'
  destruct (In_dec HA_dec a l').
  left.
  unfold incl in ×.
  intros.
  destruct H; subst; auto.
  right.
  intro.
  destruct n.
  apply H.
  simpl.
  left; auto.
~ incl l l'
  right.
  intro.
  destruct n.
  unfold incl in ×.
  intros.
  apply H.
  right; auto.
Qed.

Hint Resolve incl_dec: baseset.
