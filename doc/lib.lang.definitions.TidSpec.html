Library lib.lang.definitions.TidSpec

Require Import Arith.

Module tidspec.

  Definition A := nat.

  Definition beq : A → A → bool := beq_nat.


  Definition blt : A → A → bool := Nat.ltb.

  Lemma beq_true_eq : ∀ a a' : A,
                        beq a a' = true → a = a'.
  Proof.
    intros.
    apply beq_nat_true.
    auto.
  Qed.

  Lemma beq_false_neq : ∀ a a' : A,
                          beq a a' = false → a ≠ a'.
  Proof.
    intros.
    apply beq_nat_false.
    auto.
  Qed.


  Lemma eq_beq_true : ∀ a a' : A,
    a = a' → beq a a' = true.
  Proof.
    apply beq_nat_true_iff.
  Qed.


  Lemma neq_beq_false : ∀ a a' : A,
                          a ≠ a' → beq a a' = false.
  Proof.
    apply beq_nat_false_iff.
  Qed.

  Lemma blt_trans : ∀ a a' a'' : A,
                      blt a a' = true → blt a' a'' = true → blt a a'' = true.
  Proof.
    intros.
    rewrite Nat.ltb_lt in ×.
    eapply lt_trans; eauto.
Qed.

  Lemma blt_irrefl : ∀ a : A,
                       blt a a = false.
  Proof.
    intros.
    induction a.
    unfold blt.
    unfold Nat.ltb.
    unfold Nat.leb.
    auto.

    unfold blt.
    unfold Nat.ltb.
    auto.
  Qed.

  Lemma blt_asym : ∀ a b : A,
                      blt a b = true → blt b a = false.
  Proof.
    induction a; induction b; intro.

    unfold blt in H.
    unfold Nat.ltb in H.
    unfold Nat.leb in H.
    inversion H.

    unfold blt.
    unfold Nat.ltb.
    unfold Nat.leb.
    auto.

    unfold blt in H.
    unfold Nat.ltb in H.
    unfold Nat.leb in H.
    inversion H.

    unfold blt.
    unfold Nat.ltb.
    unfold Nat.leb.
    apply IHa.
    apply Nat.ltb_lt.
    apply Nat.ltb_lt in H.
    apply lt_S_n.
    auto.
  Qed.

  Lemma blt_beq_dec : ∀ a b : A,
                        {blt a b = true} + {beq a b = true} + {blt b a = true}.
  Proof.
    induction a; induction b; intros.

    - left; right; auto.

    - left; left; auto.

    - right; auto.

    - specialize (IHa b).
      destruct IHa as [[IHa1| IHa2] | IHa3].
      + left; left; auto.
      + left; right; auto.
      + right; auto.
  Qed.
End tidspec.
