Library lib.lemmas.Base

Require Import Omega.
Require Import Coq.omega.OmegaTactic.

Lemma nat_ind2:
  ∀ P
         (Hbase: P (O))
         (Hind: ∀ n, (∀ m, m ≤ n → P m) → P (S n)),
    ∀ n, P n.
Proof.
  intros.
  generalize dependent n.

  assert (∀ n, ∀ m, m ≤ n → P m).
  {
    induction n.
    - intros m H.
      assert (m=0) by omega.
      subst; auto.
    - intros m H.
      destruct m.
      + trivial.
      + apply Hind.
        intros.
        apply IHn.
        omega.
  }

  intros.
  eauto.
Qed.
