Lemma TSim_Seq:
  ∀ rho envs pc envp' t l ac1 ac2 ac1' ac2' envp'' i pst pst'
         (HTTerm1: ∀ sst0 : SharedState, ThreadTerm t rho (ASeq ac1 ac2) pst sst0)
         (HTTerm2: ∀ sst0 : SharedState, ThreadTerm t rho (ASeq ac1' ac2') pst' sst0)
         (HTSim1: TSim ac1 ac1' pst pst' t l pc envp' rho envs i)
         (HTSim2 : ∀ pst pst' : PrivateState,
                     PStateApprox pst pst' l pc envp' →
                     (∀ sst0 : SharedState, ThreadTerm t rho ac2 pst sst0) →
                     (∀ sst'0 : SharedState, ThreadTerm t rho ac2' pst' sst'0) →
                     
                     
                     TSim ac2 ac2' pst pst' t l pc envp'' rho envs i),
    TSim (ASeq ac1 ac2) (ASeq ac1' ac2') pst pst' t l pc envp'' rho envs i.
Proof.
  cofix Hc.

  intros.
  apply TSimProceed.
  split.
  - intros.
    match goal with [H: SharedStateApprox _ _ _ _ _ _ |- _] ⇒ rename H into HSStateApprox end.
    match goal with [H: SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep end.
    match goal with [H: seqstateII _ _ _ |- _] ⇒ rename H into HseqstII' end.
    match reverse goal with [H: seqstateII _ _ _ |- _] ⇒ rename H into HseqstII end.
    inversion HSeqStep; clear HSeqStep.
    + subst.
      match goal with [H: SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep end.
      destruct HTSim1 as [HTSim11 HTSim12].
      generalize HSeqStep; intro HSeqStep_aux.
      eapply HTSim11 in HSeqStep; eauto.
      destruct HSeqStep as [ac1'1 [pst1' [sst1' [HstarSeqStep [HSStateApprox' [HTSim11' [HseqstII1 HseqstII1']]]]]]].
      ∃ (ASeq ac1'1 ac2'), pst1', sst1'.
      split.
      × eapply starSeqStep_ASeq_starSeqStep; eauto.
      × split; auto.
        split; auto.
        assert (∀ sst0 : SharedState, ThreadTerm t rho (ASeq ac1'0 ac2) pst1 sst0). {
          eapply SeqStep_TTerm with (ac:=ASeq ac1 ac2); eauto.
          econstructor; eauto.
        }
        assert (∀ sst0 : SharedState, ThreadTerm t rho (ASeq ac1'1 ac2') pst1' sst0). {
          eapply starSeqStep_TTerm with (ac:=ASeq ac1' ac2'); eauto.
          eapply starSeqStep_ASeq_starSeqStep; eauto.
        }
        eapply Hc; eauto.

    + subst.
      destruct HTSim1 as [HTSim11 HTSim12].
      destruct HTSim11 as [HTSim11 HTSim12].
      eapply HTSim12 in HSStateApprox; auto.
      destruct HSStateApprox as [pst1' [sst1' [HstarSeqStep [HSStateApprox' [HPStateApprox' [HseqstII1 HseqstII1']]]]]].
      ∃ ac2', pst1', sst1'.
      split; auto.
      assert (starSeqStep t rho (ASeq ac1' ac2', (pst', sst')) (ASeq ASkip ac2', (pst1', sst1'))).
      {
        eapply starSeqStep_ASeq_starSeqStep; eauto.
      }
      assert (starSeqStep t rho (ASeq ASkip ac2', (pst1', sst1')) (ac2', (pst1', sst1'))).
      {
        ∃ (S O).
        simpl.
        eapply ex_intro.
        split; eauto.
        eapply SSSeqSkip; eauto.
      }
      eapply starSeqStep_trans; eauto.
      split; auto.
      split; auto.
      {
        assert (∀ sst0 : SharedState, ThreadTerm t rho ac0 pst1 sst0). {
          eapply Seq_TTerm_TTerm'; eauto.
          ∃ O.
          simpl.
          eauto.
        }
        assert (∀ sst'0 : SharedState, ThreadTerm t rho ac2' pst1' sst'0). {
          eapply Seq_TTerm_TTerm'; eauto.
        }
        eapply HTSim2; eauto.
      }
      eauto.
      split ;auto.
      split; auto.
  - intros sst sst' Hfalse.
    inversion Hfalse.

    Grab Existential Variables. eauto.
Qed.
