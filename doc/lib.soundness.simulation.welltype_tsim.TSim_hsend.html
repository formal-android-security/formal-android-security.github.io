Lemma Hsend_SStateApprox_nlabelleq:

         (HTTerm: ∀ sst0, ThreadTerm t rho (AHsend e1 e2 l') pst' sst0)

  destruct sst' as [[[ss' hqs'] lqs'] alpha'].

        destruct (HMSGSMod.eq_dec i1 t0).
        -- subst i1.
           rewrite HMSGSMod.set_a_get_a; eauto with basemap.
           eapply ex_intro.
           split; eauto.
           unfold LeakFree.HMsgqueueApprox.
           destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.
        -- rewrite HMSGSMod.set_a_get_a'; eauto with basemap.
    + intros t0 hq2 Hmsgget'.
      destruct (HMSGSMod.eq_dec i1 t0).
      × subst i1.
        match goal with [H: Some _ = HMSGSMod.get _ _ |- _] ⇒ rename H into Hmsgget end.
        apply eq_sym in Hmsgget.
        specialize (HSStateApprox22 t0 hqi0 Hmsgget).
        destruct HSStateApprox22 as [hq1 [Hmsgget1 HMsgqueueApprox]].
        rewrite HMSGSMod.set_a_get_a in Hmsgget'; eauto with basemap.
        destruct (HMSGSMod.eq_dec i0 t0).
        -- subst i0.
           rewrite HMSGSMod.set_a_get_a; eauto with basemap.
           eapply ex_intro.
           split; eauto.
           unfold LeakFree.HMsgqueueApprox.
           destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.
        -- rewrite HMSGSMod.set_a_get_a'; eauto with basemap.
           eapply ex_intro.
           split; eauto.
           unfold LeakFree.HMsgqueueApprox.
           destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.
      × rewrite HMSGSMod.set_a_get_a' in Hmsgget'; eauto with basemap.
        destruct (HMSGSMod.eq_dec i0 t0).
        -- subst i0.
           rewrite HMSGSMod.set_a_get_a; eauto with basemap.
           eapply ex_intro.
           split; eauto.
           unfold LeakFree.HMsgqueueApprox.
           destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.
        -- rewrite HMSGSMod.set_a_get_a'; eauto with basemap.
  - subst.
    match goal with [H: (_,_,_,_) = (_,_,_,_) |- _] ⇒ inversion H; clear H; subst end.
    match goal with [H: (_,_,_,_) = (_,_,_,_) |- _] ⇒ inversion H; clear H; subst end.
    split; auto.
    split; auto.
    destruct HSStateApprox2 as [HSStateApprox21 HSStateApprox22].
    assert (¬ labelleq (gap (rho i0)) l). {
      eauto with baselabel.
    }
    split.
    + intros t0 hq1 Hmsgget'.
      destruct (HMSGSMod.eq_dec i0 t0).
      × subst i0.
        match goal with [H: Some _ = HMSGSMod.get _ _ |- _] ⇒ rename H into Hmsgget end.
        apply eq_sym in Hmsgget.
        specialize (HSStateApprox21 t0 hqi Hmsgget).
        destruct HSStateApprox21 as [hq2 [Hmsgget2 HMsgqueueApprox]].
        rewrite HMSGSMod.set_a_get_a in Hmsgget'; eauto with basemap.
        eapply ex_intro.
        split; eauto.
        unfold LeakFree.HMsgqueueApprox.
        destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.
      × rewrite HMSGSMod.set_a_get_a' in Hmsgget'; eauto with basemap.
    + intros t0 hq2 Hmsgget'.
      destruct (HMSGSMod.eq_dec i0 t0).
      × subst i0.
        match goal with [H: Some _ = HMSGSMod.get _ _ |- _] ⇒ rename H into Hmsgget end.
        apply eq_sym in Hmsgget.
        specialize (HSStateApprox21 t0 hqi Hmsgget).
        destruct HSStateApprox21 as [hq1 [Hmsgget1 HMsgqueueApprox]].
        rewrite HMSGSMod.set_a_get_a; eauto with basemap.
        eapply ex_intro.
        split; eauto.
        unfold LeakFree.HMsgqueueApprox.
        destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.

    assert (¬ labelleq (gap (rho i1)) l). {
      eauto with baselabel.
    }
    split.
    + intros t0 hq1 Hmsgget'.
      destruct (HMSGSMod.eq_dec i1 t0).
      × subst i1.
        match goal with [H: Some _ = HMSGSMod.get _ _ |- _] ⇒ rename H into Hmsgget end.
        apply eq_sym in Hmsgget.
        specialize (HSStateApprox22 t0 hqi Hmsgget).
        destruct HSStateApprox22 as [hq2 [Hmsgget2 HMsgqueueApprox]].
        rewrite HMSGSMod.set_a_get_a; eauto with basemap.
        eapply ex_intro.

        destruct (labelleq_dec (gap (rho t0)) l); try congruence; auto.

      destruct (HMSGSMod.eq_dec i1 t0).
      × subst i1.
        match goal with [H: Some _ = HMSGSMod.get _ _ |- _] ⇒ rename H into Hmsgget end.

        specialize (HSStateApprox22 t0 hqi Hmsgget).
        destruct HSStateApprox22 as [hq1 [Hmsgget1 HMsgqueueApprox]].
        rewrite HMSGSMod.set_a_get_a in Hmsgget'; eauto with basemap.

Lemma Hsend_TSim_Skip_nlabelleq:
  ∀ t rho e1 e2 l' pst sst pst' sst' ac1 ac1' pst1 sst1 pst1' sst1' envs l i pc envp
         (Hnlabelleq : ¬ labelleq l' l)
         (HPStateApprox: PStateApprox pst pst' l pc envp)
         (HSeqStep : SeqStep t rho (AHsend e1 e2 l', (pst, sst)) (ac1, (pst1, sst1)))
         (HSeqStep' : SeqStep t rho (AHsend e1 e2 l', (pst', sst')) (ac1', (pst1', sst1'))),
  TSim ac1 ac1' pst1 pst1' t l pc envp rho envs i.
Proof.
  intros.
  destruct pst as [tpc ps].
  destruct pst' as [tpc' ps'].
  inversion HSeqStep; inversion HSeqStep'; eapply TSim_Skip; eauto.
Qed.

Lemma WellType_imply_TSim_HSend:
  ∀ rho t l e1 e2 pc envs envp l1 l2 i pst pst'
         (HTTerm: ∀ sst0, ThreadTerm t rho (AHsend e1 e2 (lunion (lunion pc l1) l2)) pst' sst0)
         (HexprTypeSafe1 : ExprTypeSafe envp e1 l1)
         (HexprTypeSafe2 : ExprTypeSafe envp e2 l2)
         (HPStateApprox : PStateApprox pst pst' l pc envp),
    TSim (AHsend e1 e2 (lunion (lunion pc l1) l2)) (AHsend e1 e2 (lunion (lunion pc l1) l2)) pst pst' t l pc envp rho envs i.
Proof.
  intros.
  constructor.
  split.
  - intros sst sst' ac1 pst1 sst1 HSStateApprox HSeqStep HseqstII HseqstII'.
    assert (seqstateII (pst1, sst1) t rho). {
      eapply SeqStep_seqstateII; eauto.
    }
    destruct (labelleq_dec (lunion (lunion pc l1) l2) l).
    + inversion HSeqStep.
      × subst sst1.
        subst.
        destruct sst' as [[[ss' hqs'] lqs'] alpha'].
        destruct pst' as [tpc' ps'].
        remember (ExprInterp e1 ps') as v.
        apply eq_sym in Heqv.
        destruct v as [tsi' i0'].
        remember (ExprInterp e2 ps') as v'.
        apply eq_sym in Heqv'.
        destruct v' as [ts' n'].
        assert (labelleq l1 l) by eauto with baselabel.
        assert (labelleq l2 l) by eauto with baselabel.
        assert (labelleq pc l) by eauto with baselabel.
        destruct HPStateApprox as [HPStateApprox1 HPStateApprox2].
        assert (i0 = i0' ∧ ts_eq tsi tsi') as Heqv1. {
          eapply exprInterp_eq; eauto.
        }
        destruct Heqv1.
        assert (n = n' ∧ ts_eq ts ts') as Heqv2. {
          eapply exprInterp_eq; eauto.
        }
        destruct Heqv2.
        assert (tpc = tpc') by auto.
        unfold ts_eq in ×.
        subst i0' tsi' n' ts' tpc'; subst.
        destruct HSStateApprox as [HSStateApprox1 [HSStateApprox2 [HSStateApprox3 HSStateApprox4]]].
        destruct HSStateApprox2 as [HSStateApprox21 HSStateApprox22].
        assert (∃ hqi', HMSGSMod.get hqs' i0 = Some hqi' ∧ HMsgqueueApprox hqi hqi' l (gap (rho i0))) as Hhmsgget. {
          eauto.
        }
        destruct Hhmsgget as [hqi' [Hhmsgget HMsgqueueApprox]].
        assert (SeqStep t rho (AHsend e1 e2 (lunion (lunion pc l1) l2), (tpc, ps', (ss', hqs', lqs', alpha'))) (ASkip, (tpc, ps', (ss', HMSGSMod.set hqs' i0 (hqi' ++ (t, ts_union ts (tpc_union tpc), n) :: nil), lqs', alpha')))). {
          econstructor; eauto.
        }
        assert (seqstateII (tpc, ps',
         (ss', HMSGSMod.set hqs' i0 (hqi' ++ (t, ts_union ts (tpc_union tpc), n) :: nil),
          lqs', alpha')) t rho). {
          eapply SeqStep_seqstateII; eauto.
        }
        ∃ ASkip, (tpc, ps'),
               (ss',
               HMSGSMod.set hqs' i0 (hqi' ++ (t, ts_union ts (tpc_union tpc), n) :: nil),
               lqs', alpha').
        split.
        -- ∃ (S O).
           simpl.
           eapply ex_intro.
           split; eauto.
        -- split.
           ++ split; auto.
              split; auto.
              split.
              ** intros t0 hq1 Hmsgget.
                 destruct (HMSGSMod.eq_dec i0 t0).
                 --- subst i0.
                     rewrite HMSGSMod.set_a_get_a in *; auto with basemap.
                     eapply ex_intro.
                     split; eauto.
                     inversion Hmsgget.
                     subst.
                     unfold LeakFree.HMsgqueueApprox in ×.
                     destruct (labelleq_dec (gap (rho t0)) l).
                     +++ congruence.
                     +++ auto.
                 --- rewrite HMSGSMod.set_a_get_a' in *; auto with basemap.
              ** intros t0 hq2 Hmsgget.
                 destruct (HMSGSMod.eq_dec i0 t0).
                 --- subst i0.
                     rewrite HMSGSMod.set_a_get_a in *; auto with basemap.
                     eapply ex_intro.
                     split; eauto.
                     inversion Hmsgget.
                     subst.
                     unfold LeakFree.HMsgqueueApprox in ×.
                     destruct (labelleq_dec (gap (rho t0)) l).
                     +++ congruence.
                     +++ auto.
                 --- rewrite HMSGSMod.set_a_get_a' in *; auto with basemap.
           ++ split; auto.
              eapply TSim_Skip; eauto.
      × subst sst1.
        subst.
        destruct sst' as [[[ss' hqs'] lqs'] alpha'].
        destruct pst' as [tpc' ps'].
        remember (ExprInterp e1 ps') as v.
        apply eq_sym in Heqv.
        destruct v as [tsi' i0'].
        assert (labelleq l1 l) by eauto with baselabel.
        destruct HPStateApprox as [HPStateApprox1 HPStateApprox2].
        assert (i0 = i0' ∧ ts_eq tsi tsi') as Heqv1. {
          eapply exprInterp_eq; eauto.
        }
        destruct Heqv1.
        subst i0'; subst.
        generalize HSStateApprox; intro HSStateApprox_aux.
        destruct HSStateApprox as [HSStateApprox1 [HSStateApprox2 [HSStateApprox3 HSStateApprox4]]].
        destruct HSStateApprox2 as [HSStateApprox21 HSStateApprox22].
        assert (None = HMSGSMod.get hqs' i0). {
          remember (HMSGSMod.get hqs' i0) as v.
          destruct v; auto.
          apply eq_sym in Heqv0.
          eapply HSStateApprox22 in Heqv0; eauto.
          destruct Heqv0 as [hq1 [Hhmsgget HMsgqueueApprox]].
          congruence.
        }
        ∃ ASkip, (tpc', ps'), (ss', hqs', lqs', alpha').
        split; auto.
        -- ∃ (S O).
           simpl.
           eapply ex_intro; eauto.
           split; eauto.
           eapply SSHsendfail; eauto.
        -- split; auto.
           split; auto.
           eapply TSim_Skip; eauto.
    + generalize HTTerm; intro HTTerm_aux.
      specialize (HTTerm sst').
      apply TTerm_starSeqStep_Skip in HTTerm.
      destruct HTTerm as [pst'0 [sst'0 HstarSeqStep]].
      destruct HstarSeqStep as [k HkSeqStep].
      destruct k.
      × simpl in ×.
        congruence.
      × simpl in ×.
        destruct HkSeqStep as [sc'' [HSeqStep' HkSeqStep]].
        destruct sc'' as [ac'' [pst'' sst'']].
        ∃ ac'', pst'', sst''.
        split.
        -- ∃ (S O).
           simpl.
           eapply ex_intro.
           split; eauto.
        -- split.
           ++ eapply Hsend_SStateApprox_nlabelleq; eauto.
           ++ split.
              ** eapply Hsend_TSim_Skip_nlabelleq; eauto.
              ** split; auto.
                 eapply SeqStep_seqstateII; eauto.

  - intros.
    congruence.
Qed.
