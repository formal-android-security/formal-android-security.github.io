Lemma WellType_imply_TSim_Assign:
  ∀ x e pst pst' (rho : Tid → Policy) (envs : Env) (pc : Label)
         (envp envp' : Env) (t i : Tid)
         (l : Label)
         (HTypeSafe : TypeSafe (pc, rho t, envs) envp (Assign x e) envp')
         (HPStateApprox : PStateApprox pst pst' l pc envp),
         
         
    TSim (AAssign x e) (AAssign x e) pst pst' t l pc envp' rho envs i.
Proof.
  intros.
  inversion HTypeSafe.
  subst.
  apply TSimProceed.
  split.
  - intros sst sst' ac1 pst1 sst1 HSStateApprox HSeqStep HseqstII HseqstII'.
    assert (seqstateII (pst1, sst1) t rho). {
      eapply SeqStep_seqstateII; eauto.
    }
    inversion HSeqStep.
    rename H3 into Hexprinterp.
    subst sst1.
    subst.
    destruct pst' as [tpc' ps'].
    remember (ExprInterp e ps') as v.
    apply eq_sym in Heqv.
    destruct v as [ts' n'].
    rename Heqv into Hexprinterp'.
    assert (SeqStep t rho (AAssign x e, (tpc', ps', sst')) (ASkip, (tpc', updateps ps' x (ts_add t (ts_union ts' (tpc_union tpc')), n'), sst'))) as HSeqStep'. {
      eapply SSAssign ; eauto.
    }
    assert (seqstateII
              (tpc', updateps ps' x (ts_add t (ts_union ts' (tpc_union tpc')), n'),
               sst') t rho). {
      eapply SeqStep_seqstateII; eauto.
    }
    ∃ ASkip,
                             (tpc', updateps ps' x (ts_add t (ts_union ts' (tpc_union tpc')), n')),
                             sst'.
    split.
    ∃ (S O); simpl.
    eapply ex_intro.
    split; eauto.
    split; eauto with basesound.
    split; auto.
    eapply TSim_Skip.
    remember (updateenv envp x (lunion pc l0)) as envp'.
    destruct HPStateApprox as [HPStateApprox1 HPStateApprox2].
    destruct (labelleq_dec (envp' x) l) as [Hlabelleq | Hnlabelleq].
    + assert (labelleq pc l). {
        subst.
        unfold updateenv in Hlabelleq.
        destruct (Nat.eq_dec x x); try congruence.
        eauto with baselabel.
      }
      assert (labelleq l0 l). {
        subst.
        unfold updateenv in Hlabelleq.
        destruct (Nat.eq_dec x x); try congruence.
        eauto with baselabel.
      }
      assert (tpc = tpc'). {
        eauto.
      }
      assert (n = n' ∧ ts_eq ts ts') as Hveq. {
        eapply exprInterp_eq; eauto.
      }
      destruct Hveq as [Hveq1 Hveq2].
      eapply updatepst_approx_labelleq; eauto with basesound.
    + eapply updatepst_approx_nlabelleq; eauto.
  - intros.
    congruence.
Qed.
