Compositionality states that if each process in the program P simulates the process with the same process identifer in program P', then P simulates P'.
With Compositionality, we can decompose the proof of the simulation relation between two programs into the proof of the simulation relation between two commands containing in these two programs. i.e. if for each i, Ci ≼ Ci', then P ≼ P', where P = (C1 || ... || Cn) and P' = (C1' || ... || Cn').
The proof of Compositionality follows the steps below:
Since TSim and PSim contain two cases, we prove the compositionality of these two cases seperately:
  • All processes in the program P are ASkip, i.e. P = (ASkip || ... || ASkip):
    This case is proved in Lemma PETSimSkip_imply_PSimSkip. PETSimSkip states that for each i, ASkip ≼ Ci' and PSimSkip states that P ≼ P'.
    To prove that P' can execute to ASkip, we arrange the execution of P' as C1' --> C2' --> ... -> Cn', which means that P' executes C1' to skip firstly, then continues to execute C2' to skip, and then executes the remaining commands in the order of the process identifers, until all commands executes to be skip.
    The concrete technique is by the induction of the numbers of commands in P.
  • The program P executes one step further.
    By coinduction of P ≼ P'.
    If P executes one step, then exists i such that Ci executes one step. Since Ci ≼ Ci', we know that Ci' executes one step properly. So P' executes one step properly. Then by coinduction hypothesis, we know that P ≼ P'.

Definition TSimSkip(ac ac' : AnnoCmd) (pst pst' : PrivateState) (t : Tid) (l pc : Label)
           (envp' : Env) (rho : Policys) (envs : Env) (i : Tid) :=
  ∀ sst sst' : SharedState,
    ac = ASkip →
    SharedStateApprox sst sst' i l rho envs →
    
    (seqstateII (pst', sst') t rho) →

    ∃ (pst1' : PrivateState) (sst1' : SharedState),
      starSeqStep t rho (ac', (pst', sst'))
                  (ASkip, (pst1', sst1')) ∧
      SharedStateApprox sst sst1' i l rho envs ∧
      PStateApprox pst pst1' l pc envp' ∧
      seqstateII (pst1', sst1') t rho.

Definition PETSimSkip (pe pe': Progexe) (psts psts': PrivateStateSet) l pcs envps' rho envs i :=
  ∀ (t : tidspec.A) (ac : progexespec.B) (pst : pstsspec.B),
    PEMod.get pe t = Some ac →
    PSTSMod.get psts t = Some pst →
    ∃ (ac' : progexespec.B) (pst' : pstsspec.B),
      PSTSMod.get psts' t = Some pst' ∧
      PEMod.get pe' t = Some ac' ∧
      TSimSkip ac ac' pst pst' t l (pcs t) (envps' t) rho envs i.

Definition PSimSkip(pe pe' : Progexe) (psts psts' : PrivateStateSet) (l: Label) (pcs: PC_Map)
           (envps' : Env_Map) (rho : Policys) (envs : Env) (i : Tid) :=
  ∀ sst sst' : SharedState,
    isPSkip pe →
    SharedStateApprox sst sst' i l rho envs →
    
    stateII (psts', sst') rho →
    ∃ (psts1' : PrivateStateSet) (sst1' : SharedState),
      starParaStep rho (pe', (psts', sst'))
                   (peSkip pe', (psts1', sst1')) ∧
      SharedStateApprox sst sst1' i l rho envs ∧
      PStatesApprox psts psts1' l pcs envps' ∧
      stateII (psts1', sst1') rho.

Lemma isPSkip_merge_isPSkip_l:
  ∀ pe1 pe2
         (HisPSkip: isPSkip (PEMod.merge pe1 pe2)),
    isPSkip pe1.
Proof.
  intros.
  unfold isPSkip.
  intros i ac Hget.
  unfold isPSkip in HisPSkip.
  specialize (HisPSkip i ac).
  rewrite PEMod.merge_sem in HisPSkip.
  rewrite Hget in HisPSkip.
  destruct (PEMod.get pe2 i); auto.
Qed.

Lemma isPSkip_merge_isPSkip_r:
  ∀ pe1 pe2
         (Hdisj: PEMod.disj pe1 pe2)
         (HisPSkip: isPSkip (PEMod.merge pe1 pe2)),
    isPSkip pe2.
Proof.
  intros.
  unfold isPSkip.
  intros i ac Hget.
  unfold isPSkip in HisPSkip.
  unfold PEMod.disj in Hdisj.
  specialize (Hdisj i).
  specialize (HisPSkip i ac).
  rewrite PEMod.merge_sem in HisPSkip.
  rewrite Hget in ×.
  destruct (PEMod.get pe1 i); [exfalso;auto | auto].
Qed.

Lemma PStatesApprox_get_None_get_None:
  ∀ l psts1 psts2 pcs envps t
         (HPStatesApprox: PStatesApprox psts1 psts2 l pcs envps)
         (Hget: PSTSMod.get psts1 t = None),
    PSTSMod.get psts2 t = None.
Proof.
  intros.
  destruct HPStatesApprox as [HPStatesApprox1 HPStatesApprox2].
  destruct (PSTSMod.get_dec psts2 t) as [Hget' | Hget'].

  destruct Hget' as [pst2 Hget'].
  apply HPStatesApprox2 in Hget'.
  destruct Hget' as [pst1 [Hget' _]].
  congruence.

  auto.
Qed.

Lemma PStatesApprox_merge:
  ∀ psts1 psts2 psts1' psts2' i l envps
         (HPStatesApprox1: PStatesApprox psts1 psts1' i l envps)
         (HPStatesApprox2: PStatesApprox psts2 psts2' i l envps),
    PStatesApprox (PSTSMod.merge psts1 psts2) (PSTSMod.merge psts1' psts2') i l envps.
Proof.
  intros.
  unfold PStatesApprox.
  split.
  - intros t pst1 Hget.
    apply PSTSMod.merge_get in Hget.
    destruct Hget as [Hget | Hget].
    + destruct HPStatesApprox1 as [HPStatesApprox11 HPStatesApprox12].
      apply HPStatesApprox11 in Hget.
      destruct Hget as [pst2 [Hget HPStateApprox]].
      ∃ pst2.
      split; auto.
      apply PSTSMod.merge_get_l; auto.
    + destruct Hget as [Hget1 Hget2].
      destruct HPStatesApprox1 as [HPStatesApprox11 HPStatesApprox12].
      destruct HPStatesApprox2 as [HPStatesApprox21 HPStatesApprox22].
      apply HPStatesApprox21 in Hget2.
      destruct Hget2 as [pst2 [Hget HPStateApprox]].
      ∃ pst2.
      split; auto.
      apply PSTSMod.merge_get_r; auto.
      eapply PStatesApprox_get_None_get_None; eauto.

  - intros t pst1' Hget.
    apply PStatesApprox_sym in HPStatesApprox1.
    apply PStatesApprox_sym in HPStatesApprox2.
    apply PSTSMod.merge_get in Hget.
    destruct Hget as [Hget | Hget].
    + destruct HPStatesApprox1 as [HPStatesApprox11 HPStatesApprox12].
      apply HPStatesApprox11 in Hget.
      destruct Hget as [pst2 [Hget HPStateApprox]].
      ∃ pst2.
      split; auto with basestate.
      apply PSTSMod.merge_get_l; auto.
    + destruct Hget as [Hget1 Hget2].
      destruct HPStatesApprox1 as [HPStatesApprox11 HPStatesApprox12].
      destruct HPStatesApprox2 as [HPStatesApprox21 HPStatesApprox22].
      apply HPStatesApprox21 in Hget2.
      destruct Hget2 as [pst2 [Hget HPStateApprox]].
      ∃ pst2.
      split; auto with basestate.
      apply PSTSMod.merge_get_r; auto.
      eapply PStatesApprox_get_None_get_None; eauto.
Qed.

Lemma ProgTerm_merge_r:
  ∀ pe1 psts1 pe2 psts2 rho
         (Hpedisj : PEMod.disj pe1 pe2)
         (Hpstsdisj : PSTSMod.disj psts1 psts2)
         (HPTerm : ProgTerm rho (PEMod.merge pe1 pe2) (PSTSMod.merge psts1 psts2)),
    ProgTerm rho pe2 psts2.
Proof.
  intros.
  unfold ProgTerm in ×.
  intros.
  eapply HPTerm; eauto.
  - assert (PSTSMod.get psts1 t = None). {
      unfold PSTSMod.disj in ×.
      specialize (Hpstsdisj t).
      remember (PSTSMod.get psts1 t) as v.
      destruct v; auto.
      destruct (PSTSMod.get psts2 t).
      + exfalso; auto.
      + congruence.
    }
    eapply PSTSMod.merge_get_r; eauto.
  - assert (PEMod.get pe1 t = None). {
      unfold PEMod.disj in ×.
      specialize (Hpedisj t).
      remember (PEMod.get pe1 t) as v.
      destruct v; auto.
      destruct (PEMod.get pe2 t).
      + exfalso; auto.
      + congruence.
    }
    eapply PEMod.merge_get_r; eauto.
Qed.

Lemma pstatesII_merge:
  ∀ psts1 psts2 rho
         (HpstsII2 : pstatesII psts2 rho )
         (HpstsII1 : pstatesII psts1 rho),
    pstatesII (PSTSMod.merge psts1 psts2) rho.
Proof.
  intros.
  unfold pstatesII in ×.

  remember (PSTSMod.get psts1 t) as Hgetpsts1.
  remember (PSTSMod.get psts2 t) as Hgetpsts2.
  apply eq_sym in HeqHgetpsts1.
  apply eq_sym in HeqHgetpsts2.
  destruct Hgetpsts1;destruct Hgetpsts2.
  - assert (b = pst) by congruence.
    subst.
    auto.
  - assert (b = pst) by congruence.
    subst.
    auto.
  - assert (b = pst) by congruence.
    subst.
    auto.
  - congruence.
Qed.

Lemma PSimSkip_comp:
  ∀ pe1 pe1' psts1 psts1' pe2 pe2' psts2 psts2' pcs envps' rho envs i l
         (HPSimSkip1: PSimSkip pe1 pe1' psts1 psts1' l pcs envps' rho envs i)
         (HPSimSkip2: PSimSkip pe2 pe2' psts2 psts2' l pcs envps' rho envs i)
         (Hpedisj1: PEMod.disj pe1 pe2)
         (Hpedisj2: PEMod.disj pe1' pe2')
         (Hpstsdisj1: PSTSMod.disj psts1 psts2)
         (Hpstsdisj2: PSTSMod.disj psts1' psts2') ,
    PSimSkip (PEMod.merge pe1 pe2) (PEMod.merge pe1' pe2') (PSTSMod.merge psts1 psts2) (PSTSMod.merge psts1' psts2') l pcs envps' rho envs i.
Proof.
  intros.
  unfold PSimSkip.
  intros sst sst' HisPSkip HSStateApprox HsstII.

  assert (isPSkip pe1) as HisPSkip1 by (eapply isPSkip_merge_isPSkip_l; eauto).
  assert (isPSkip pe2) as HisPSkip2 by (eapply isPSkip_merge_isPSkip_r; eauto).

  assert (stateII (psts1', sst') rho) as HsstII1. {
    eapply stateII_merge_l; eauto.
  }

  assert (stateII (psts2', sst') rho) as HsstII2. {
    eapply stateII_merge_r; eauto.
  }

  unfold PSimSkip in HPSimSkip1.
  specialize (HPSimSkip1 sst sst' HisPSkip1 HSStateApprox HsstII1).
  destruct HPSimSkip1 as [psts1'' [sst1'' [HstarParaStep1 [HSStateApprox1 [HPStatesApprox1 HsstII1'] ]]]].

  assert (starParaStep rho
                       (PEMod.merge pe1' pe2', (PSTSMod.merge psts1' psts2', sst'))
                       (PEMod.merge (peSkip pe1') pe2', (PSTSMod.merge psts1'' psts2', sst1'')) ∧
          SharedStateApprox sst sst1'' i l rho envs) as HstarParaStep'
      by (split; auto; apply starParaStep_merge_starParaStep_l; auto).
  destruct HstarParaStep' as [HstarParaStep' HSStateApprox'].

  assert (SharedStateApprox sst sst1'' i l rho envs) as HSStateApprox1'
      by (auto with basestate) .
  unfold PSimSkip in HPSimSkip2.
  assert (stateII (psts2', sst1'') rho) as HsstII2'. {
    split.
    - destruct HsstII2; auto.
    - destruct HsstII1'; auto.
  }
  
  specialize (HPSimSkip2 sst sst1'' HisPSkip2 HSStateApprox1' HsstII2').
  destruct HPSimSkip2 as [psts2'' [sst2'' [HstarParaStep2 [HSStateApprox2 [HPStatesApprox2 HsstII2'']]]]].

  assert (
      starParaStep rho
                   (PEMod.merge (peSkip pe1') pe2', (PSTSMod.merge psts1'' psts2', sst1''))
                   ((PEMod.merge (peSkip pe1') (peSkip pe2')), (PSTSMod.merge psts1'' psts2'', sst2'')) ∧
      SharedStateApprox sst sst2'' i l rho envs) as HstarParaStep''.
  split; eauto with basestate.
  apply starParaStep_merge_starParaStep_r; auto.
  unfold peSkip.
  apply PEMod.disj_mapmap_disj; auto.
  apply PSTSMod.disj_sym.
  apply PSTSMod.disj_sym in Hpstsdisj2.
  eapply starParaStep_psts_disj_preserve; eauto.
  destruct HstarParaStep'' as [HstarParaStep'' HSStateApprox''].

  assert (starParaStep rho
                       (PEMod.merge pe1' pe2', (PSTSMod.merge psts1' psts2', sst'))
                       (PEMod.merge (peSkip pe1') (peSkip pe2'), (PSTSMod.merge psts1'' psts2'', sst2'')))
    as HstarParaStep
      by (eapply starParaStep_trans; eauto).

  ∃ (PSTSMod.merge psts1'' psts2''), sst2''.
  split; auto.
  unfold peSkip.
  rewrite PEMod.mapmap_merge_eq. auto.
  split; auto.
  split.
  apply PStatesApprox_merge; auto.
  {
    destruct HsstII2''.
    destruct HsstII1'.
    split; auto.
    eapply pstatesII_merge; eauto.
  }
Qed.

Lemma PETSimSkip_partition_l:
  ∀ pe pe1 pe2 pe' pe1' pe2' psts psts1 psts2 psts' psts1' psts2' l pcs envps' rho envs i
         (Hpstseqdom2: PSTSMod.eqdom psts2 psts2')
         (Hpeeqdom2: PEMod.eqdom pe2 pe2')
         (Hpejoin: PEMod.join pe1 pe2 pe)
         (Hpejoin': PEMod.join pe1' pe2' pe')
         (Hpstsjoin: PSTSMod.join psts1 psts2 psts)
         (Hpstsjoin': PSTSMod.join psts1' psts2' psts')
         (HPESimSkip: PETSimSkip pe pe' psts psts' l pcs envps' rho envs i),
    PETSimSkip pe1 pe1' psts1 psts1' l pcs envps' rho envs i.
Proof.
  intros.
  unfold PETSimSkip in ×.
  intros t ac pst Hgetpe1 Hgetpsts1.
  assert (PEMod.get pe t = Some ac) as Hgetpe
      by (eapply PEMod.join_get_get_l; eauto).
  assert (PSTSMod.get psts t = Some pst) as Hgetpsts
      by (eapply PSTSMod.join_get_get_l; eauto).
  specialize (HPESimSkip t ac pst Hgetpe Hgetpsts).
  destruct HPESimSkip as [ac' [pst' [Hgetpsts' [Hgetpe' HTSimSkip]]]].

  ∃ ac', pst'.
  split; auto.
  assert (PSTSMod.disj psts1 psts2) as Hpstsdisj
      by (eapply PSTSMod.my_join_disj; eauto).
  assert (PSTSMod.get psts1' t = Some pst' ∨ PSTSMod.get psts2' t = Some pst') as Hgetpsts1'
      by (eapply PSTSMod.join_get_or; eauto).
  destruct Hgetpsts1'; auto.
  assert (∃ pst'', PSTSMod.get psts2 t = Some pst'') as Hgetpsts2
      by (unfold PSTSMod.eqdom in Hpstseqdom2;
          unfold PSTSMod.indom in Hpstseqdom2;
          eapply Hpstseqdom2; eauto).
  destruct Hgetpsts2 as [pst'' Hgetpsts2].
  unfold PSTSMod.disj in Hpstsdisj.
  specialize (Hpstsdisj t).
  rewrite Hgetpsts1 in Hpstsdisj.
  rewrite Hgetpsts2 in Hpstsdisj.
  exfalso; auto.

  split; auto.
  assert (PEMod.disj pe1 pe2) as Hpedisj
      by (eapply PEMod.my_join_disj; eauto).
  assert (PEMod.get pe1' t = Some ac' ∨ PEMod.get pe2' t = Some ac') as Hgetpe1'
      by (eapply PEMod.join_get_or; eauto).
  destruct Hgetpe1'; auto.
  assert (∃ ac'', PEMod.get pe2 t = Some ac'') as Hgetpe2
      by (unfold PEMod.eqdom in Hpeeqdom2;
          unfold PEMod.indom in Hpeeqdom2;
          eapply Hpeeqdom2; eauto).
  destruct Hgetpe2 as [ac'' Hgetpe2].
  unfold PEMod.disj in Hpedisj.
  specialize (Hpedisj t).
  rewrite Hgetpe1 in Hpedisj.
  rewrite Hgetpe2 in Hpedisj.
  exfalso; auto.
Qed.

Lemma PETSimSkip_partition_r:
  ∀ pe pe1 pe2 pe' pe1' pe2' psts psts1 psts2 psts' psts1' psts2' l pcs envps' rho envs i
         (Hpstseqdom1: PSTSMod.eqdom psts1 psts1')
         (Hpeeqdom1: PEMod.eqdom pe1 pe1')
         (Hpejoin: PEMod.join pe1 pe2 pe)
         (Hpejoin': PEMod.join pe1' pe2' pe')
         (Hpstsjoin: PSTSMod.join psts1 psts2 psts)
         (Hpstsjoin': PSTSMod.join psts1' psts2' psts')
         (HPESimSkip: PETSimSkip pe pe' psts psts' l pcs envps' rho envs i),
    PETSimSkip pe2 pe2' psts2 psts2' l pcs envps' rho envs i.
Proof.
  intros.
  unfold PETSimSkip in ×.
  intros t ac pst Hgetpe2 Hgetpsts2.
  assert (PEMod.get pe t = Some ac) as Hgetpe
      by (eapply PEMod.join_get_get_r; eauto).
  assert (PSTSMod.get psts t = Some pst) as Hgetpsts
      by (eapply PSTSMod.join_get_get_r; eauto).
  specialize (HPESimSkip t ac pst Hgetpe Hgetpsts).
  destruct HPESimSkip as [ac' [pst' [Hgetpsts' [Hgetpe' HTSimSkip]]]].

  ∃ ac', pst'.
  split; auto.
  assert (PSTSMod.disj psts1 psts2) as Hpstsdisj
      by (eapply PSTSMod.my_join_disj; eauto).
  assert (PSTSMod.get psts1' t = Some pst' ∨ PSTSMod.get psts2' t = Some pst') as Hgetpsts2'
      by (eapply PSTSMod.join_get_or; eauto).
  destruct Hgetpsts2'; auto.
  assert (∃ pst'', PSTSMod.get psts1 t = Some pst'') as Hgetpsts1
      by (unfold PSTSMod.eqdom in Hpstseqdom1;
          unfold PSTSMod.indom in Hpstseqdom1;
          eapply Hpstseqdom1; eauto).
  destruct Hgetpsts1 as [pst'' Hgetpsts1].
  unfold PSTSMod.disj in Hpstsdisj.
  specialize (Hpstsdisj t).
  rewrite Hgetpsts1 in Hpstsdisj.
  rewrite Hgetpsts2 in Hpstsdisj.
  exfalso; auto.

  split; auto.
  assert (PEMod.disj pe1 pe2) as Hpedisj
      by (eapply PEMod.my_join_disj; eauto).

Lemma PETSimSkip_imply_PSimSkip_sig:
  ∀ t ac ac' psts psts' l pcs envps' rho envs i
         (Hpepstseqdom: PEPSTSMod.eqdom (PEMod.sig t ac) psts)
         (Hpepstseqdom': PEPSTSMod.eqdom (PEMod.sig t ac') psts')
         (HPESimSkip: PETSimSkip (PEMod.sig t ac) (PEMod.sig t ac') psts psts' l pcs envps' rho envs i),
    PSimSkip (PEMod.sig t ac) (PEMod.sig t ac') psts psts' l pcs envps' rho envs i.
Proof.
  intros.
  unfold PETSimSkip in HPESimSkip.
  assert (∃ pst, PSTSMod.get psts t = Some pst) as Hgetpsts.
  unfold PEPSTSMod.eqdom in Hpepstseqdom.
  unfold PEMod.indom in ×.
  unfold PSTSMod.indom in ×.
  eapply Hpepstseqdom.
  rewrite PEMod.get_sig_some; auto.
  eauto.
  assert (PEMod.get (PEMod.sig t ac) t = Some ac) as Hgetsig
      by(eapply PEMod.get_sig_some; eauto).
  destruct Hgetpsts as [pst Hgetpsts].
  specialize (HPESimSkip t ac pst Hgetsig Hgetpsts).
  destruct HPESimSkip as [ac'0 [pst' [Hgetpsts' [Hgetsig' HTSimSkip]]]].
  assert (ac' = ac'0)
    by (rewrite PEMod.get_sig_some in Hgetsig'; congruence).
  subst ac'0.
  clear Hgetsig'.

  unfold TSimSkip in HTSimSkip.
  unfold PSimSkip.
  intros sst sst' HisPSkip HSStateApprox HsstII.
  assert (ac = ASkip) as HacSkip by eauto.
  assert (seqstateII (pst', sst') t rho) as HseqstII. {
    destruct HsstII as [HpstsII HsstII].
    split; auto.
  }
  
  specialize (HTSimSkip sst sst' HacSkip HSStateApprox HseqstII).
  destruct HTSimSkip as [pst1' [sst1' [HstarSeqStep [HSStateApprox' [HPStateApprox HseqstII']]]]].
  ∃ (updatepsts psts' t pst1'), sst1'.
  assert (∃ pst1, psts = PSTSMod.sig t pst1) as Hpstssig
      by (eapply PEPSTSMod.sig_eqdom_sig; eauto).
  destruct Hpstssig as [pst1 Hpstssig].
  assert (∃ pst'1, psts' = PSTSMod.sig t pst'1) as Hpstssig'
      by (eapply PEPSTSMod.sig_eqdom_sig; eauto).
  destruct Hpstssig' as [pst'1 Hpstssig'].
  subst psts psts'.
  rewrite PSTSMod.get_sig_some in *; auto.
  assert (pst = pst1) by congruence.
  assert (pst'1 = pst') by congruence.
  subst pst pst'1.
  split; auto.
  assert (peSkip (PEMod.sig t ac') = updatepe (PEMod.sig t ac') t ASkip) as HpeSkipeq.
  unfold peSkip.
  unfold updatepe.
  rewrite <- PEMod.set_sig_mapmap.
  rewrite PEMod.set_sig; auto.
  rewrite HpeSkipeq.
  eapply starSeqStep_imply_starParaStep; eauto.
  apply PEMod.get_sig_some.
  apply PSTSMod.get_sig_some.

  split; auto.
  unfold updatepsts.
  rewrite PSTSMod.set_sig.
  split.
  eapply PStatesApprox_sig; eauto.
  apply sstateII_sig; auto.
Qed.

Lemma PETSimSkip_imply_PSimSkip:
  ∀ pe pe' psts psts' pcs envps' rho envs i l
         (Hpeeqdom: PEMod.eqdom pe pe')
         (HPPeqdom: PEPSTSMod.eqdom pe psts)
         (HPPeqdom': PEPSTSMod.eqdom pe' psts')
         (HTSimSkip : PETSimSkip pe pe' psts psts' l pcs envps' rho envs i),
    PSimSkip pe pe' psts psts' l pcs envps' rho envs i.
Proof.
  destruct pe.
  induction lst.

  intros.
  assert (PEMod.listset_con (lst:=nil) i = PEMod.emp) as Hemp
      by (subst; eauto with basemap).
  rewrite Hemp in *; clear Hemp.
  assert (pe' = PEMod.emp) as Hemp'
      by (apply PEMod.eqdom_emp_emp_r; auto).
  rewrite Hemp' in *; clear Hemp'.
  assert (psts = PSTSMod.emp) as Hemp''
      by (apply PEPSTSMod.eqdom_emp_emp_r; auto).
  assert (psts' = PSTSMod.emp) as Hemp'''
      by (apply PEPSTSMod.eqdom_emp_emp_r; auto).
  subst.
  apply emp_PSimSkip; auto.

  intros.
  destruct a.
  destruct i.
  specialize (IHlst i).
  remember (PEMod.listset_con (lst:=(a, b) :: lst) (conj l0 i)) as pe.
  remember (PEMod.sig a b) as pe1.
  remember (PEMod.listset_con (lst:=lst) i) as pe2.
  assert (PEMod.join pe1 pe2 pe) as Hpejoin
      by (subst; eapply PEMod.con_join; eauto).
  assert (∃ pe1' pe2', PEMod.join pe1' pe2' pe' ∧ PEMod.eqdom pe1 pe1' ∧ PEMod.eqdom pe2 pe2') as Hpe'part
      by (eapply PEMod.eqdom_join_partition; eauto).
  destruct Hpe'part as [pe1' [pe2' [Hpe'join [Hpe'eqdom1' Hpe'eqdom2']]]].
  assert (pe' = PEMod.merge pe1' pe2') as Hpe'
      by (apply PEMod.join_merge; auto).
  assert (pe = PEMod.merge pe1 pe2) as Hpe
      by (apply PEMod.join_merge; auto).
  rewrite Hpe.
  rewrite Hpe'.
  assert (∃ psts1 psts2, PSTSMod.join psts1 psts2 psts ∧ PEPSTSMod.eqdom pe1 psts1 ∧ PEPSTSMod.eqdom pe2 psts2) as Hpstspart
      by (eapply PEPSTSMod.eqdom_join_partition_r; eauto).
  destruct Hpstspart as [psts1 [psts2 [Hpstsjoin [Hpstseqdom1 Hpstseqdom2]]]].

  assert (PSTSMod.eqdom psts psts') as Hpstseqdom by
        (eapply PEPSTSMod.m1_eqdom_eqdom_m2_eqdom with (m1:=pe); eauto).
  assert (∃ psts1' psts2', PSTSMod.join psts1' psts2' psts' ∧ PSTSMod.eqdom psts1 psts1' ∧ PSTSMod.eqdom psts2 psts2') as Hpsts'part
      by (eapply PSTSMod.eqdom_join_partition; eauto).
  destruct Hpsts'part as [psts1' [psts2' [Hjoin' [Heqpsts'dom1 Hpsts'eqdom2']]]].
  assert (psts' = PSTSMod.merge psts1' psts2') as Hpsts'
      by (apply PSTSMod.join_merge; auto).
  assert (psts = PSTSMod.merge psts1 psts2) as Hpsts
      by (apply PSTSMod.join_merge; auto).
  rewrite Hpsts.
  rewrite Hpsts'.
  assert (PEPSTSMod.eqdom pe2' psts2') as HPPeqdom2'
      by (eapply PEPSTSMod.eqdom_m1_eqdom_m2_eqdom_eqdom; eauto).
  assert (PETSimSkip pe2 pe2' psts2 psts2' l pcs envps' rho envs i0)
    by (eapply PETSimSkip_partition_r with (pe1:=pe1) (psts1:=psts1); eauto).
  assert (PSimSkip pe2 pe2' psts2 psts2' l pcs envps' rho envs i0)
    by eauto.
  eapply PSimSkip_comp;
    try(eapply PEMod.my_join_disj; eauto);
    try(eapply PSTSMod.my_join_disj; eauto);
    eauto.

  assert (PETSimSkip pe1 pe1' psts1 psts1' l pcs envps' rho envs i0) as HPETSimSkip1
      by (eapply PETSimSkip_partition_l with (pe2:=pe2) (psts2:=psts2); eauto).

  assert (∃ b', pe1' = PEMod.sig a b') as Hpe1'sig
      by (subst pe1; eapply PEMod.sig_eqdom_sig; eauto).
  destruct Hpe1'sig as [b' Hpe1'sig].

  assert (∃ pst, psts1 = PSTSMod.sig a pst) as Hpstssig
      by (subst pe1; eapply PEPSTSMod.sig_eqdom_sig; eauto).
  destruct Hpstssig as [pst Hpstssig].

  assert (∃ pst', psts1' = PSTSMod.sig a pst') as Hpsts'sig
      by (subst psts1; eapply PSTSMod.sig_eqdom_sig; eauto).
  destruct Hpsts'sig as [pst' Hpsts'sig].

  subst pe1 pe1' psts1 psts1'.
  eapply PETSimSkip_imply_PSimSkip_sig; eauto.
  eapply PEPSTSMod.sig_eqdom.
Qed.

Lemma Compositionality:
  ∀ pe pe' psts psts' pcs envps' rho envs i l
         (Hpepstseqdom: PEPSTSMod.eqdom pe psts)
         (Hpepstseqdom': PEPSTSMod.eqdom pe' psts')
         (Heqdom: PEMod.eqdom pe pe'),
    (
      ∀ t ac pst,
        PEMod.get pe t = Some ac →
        PSTSMod.get psts t = Some pst →
        ∃ ac' pst',
          PSTSMod.get psts' t = Some pst' ∧
          PEMod.get pe' t = Some ac' ∧
          TSim ac ac' pst pst' t l (pcs t) (envps' t) rho envs i
    )
    →
    PSim pe pe' psts psts' l pcs envps' rho envs i.
Proof.
  cofix Hc.
  intros.
  rename H into HTSim.

  intros.
  apply PSimProceed.
  split.
  - intros.
    match goal with [H:ParaStep _ _ _ |- _] ⇒ rename H into HParaStep end.
    match goal with [H:SharedStateApprox _ _ _ _ _ _ |- _] ⇒ rename H into HSStateApprox end.
    match goal with [H: stateII _ _ |- _] ⇒ rename H into HsstII2 end.
    match reverse goal with [H: stateII _ _ |- _] ⇒ rename H into HsstII1 end.

    inversion HParaStep.
    subst.
    rename sst1 into sst'0.
    rename pst' into pst'0.
    match goal with [H:SeqStep _ _ _ _ |- _] ⇒ rename H into HSeqStep end.
    match goal with [H: PEMod.get _ _ = Some _ |- _] ⇒ rename H into HGetpe end.
    match goal with [H: PSTSMod.get _ _ = Some _ |- _] ⇒ rename H into HGetpsts end.
    generalize HTSim; intro HTSim_aux.
    specialize (HTSim i0 aci pst HGetpe HGetpsts).
    destruct HTSim as [ac' [pst' [HGetpe' [HGetpsts' HTSim]]]].
    destruct HTSim as [HTSim1 HTSim2].
    destruct HTSim1 as [HTSim1 HTSim2].
    destruct HsstII1 as [HpstsII1 HsstII1].
    destruct HsstII2 as [HpstsII2 HsstII2].
    assert (seqstateII (pst, sst) i0 rho) as HseqstII. {
      split; auto.
    }
    assert (seqstateII (pst', sst') i0 rho) as HseqstII'. {
      split; auto.
    }
    
    specialize (HTSim1 sst sst' aci' pst'0 sst'0 HSStateApprox HSeqStep HseqstII HseqstII').
    destruct HTSim1 as [ac1' [pst1' [sst1' [HSeqStepStar [HSStateApprox' [HTSim' [HseqstI11 HseqstII1']]]]]]].
    ∃ (updatepe pe' i0 ac1'), (updatepsts psts' i0 pst1'), sst1'.
    split.
    + eapply starSeqStep_imply_starParaStep; eauto.
    + split; auto.
      split.
      × assert (PEPSTSMod.eqdom (updatepe pe i0 aci') (updatepsts psts i0 pst'0))
          by (
              unfold updatepe;
              unfold updatepsts;
              apply PEPSTSMod.eqdom_set_set; auto).
        assert (PEPSTSMod.eqdom (updatepe pe' i0 ac1') (updatepsts psts' i0 pst1'))
          by (
              unfold updatepe;
              unfold updatepsts;
              apply PEPSTSMod.eqdom_set_set; auto).
        assert(PEMod.eqdom (updatepe pe i0 aci') (updatepe pe' i0 ac1'))
          by (
              unfold updatepe;
              unfold updatepsts;
              apply PEMod.eqdom_set_set; auto).
        eapply Hc; eauto.
        {
          intros.
          destruct (eq_tid_dec i0 t).
          - subst i0.
            assert (PEMod.get (updatepe pe t aci') t = Some aci') by auto with basemap.
            assert (ac =aci') by congruence.
            subst aci'.
            ∃ ac1', pst1'.
            split; auto with basemap.
            split; auto with basemap.
            assert (PSTSMod.get (updatepsts psts t pst'0) t = Some pst'0) by auto with basemap.
            assert (pst'0 = pst0) by congruence.
            subst; auto.
          - assert (PEMod.get (updatepe pe i0 aci') t = PEMod.get pe t) by auto with basemap.
            assert (PEMod.get pe t = Some ac) as HGetpet by congruence.
            assert (PSTSMod.get (updatepsts psts i0 pst'0) t = PSTSMod.get psts t) by auto with basemap.
            assert (PSTSMod.get psts t = Some pst0) as HGetpstst by congruence.
            specialize (HTSim_aux t ac pst0 HGetpet HGetpstst).
            destruct HTSim_aux as [ac'' [pst'' [HGetpe'' [HGetpsts'' HTSim_aux]]]].
            ∃ ac'', pst''.
            assert (PEMod.get (updatepe pe' i0 ac1') t = PEMod.get pe' t) by auto with basemap.
            assert (PEMod.get pe' t = Some ac'') as HGetpe't by congruence.
            assert (PSTSMod.get (updatepsts psts' i0 pst1') t = PSTSMod.get psts' t) by auto with basemap.
            assert (PSTSMod.get psts' t = Some pst'') as HGetpsts't by congruence.
            split; try congruence.
            split; try congruence.
        }
      × split; eapply stateII_update; eauto.

  - intros.
    assert (PETSimSkip pe pe' psts psts' l pcs envps' rho envs i) as HPESimSkip. {
      unfold PETSimSkip.
      intros t ac pst Hgetpe Hgetpsts.
      specialize (HTSim t ac pst Hgetpe Hgetpsts).
      destruct HTSim as [ac' [pst' [Hgetpsts' [Hgetpe' HTSim]]]].
      ∃ ac', pst'.
      split; auto.
      split; auto.
      apply TSim_imply_TSimSkip; auto.
    }
    assert (PSimSkip pe pe' psts psts' l pcs envps' rho envs i) as HPSimSkip
        by (eapply PETSimSkip_imply_PSimSkip; eauto).
    auto.
Qed.
