Lemma labelleq_dec:

    {labelleq l1 l2} + {¬ labelleq l1 l2}.

  intros.
  intros.
  - destruct l1; destruct l2.
    + left; auto with baselabel.
    + right.
      intro.
      inversion H.
      × congruence.
      × destruct H0 as [les1 [les2 [H01 H02]]].
        congruence.
    + left. auto with baselabel.
    + destruct (incl_dec Vars Nat.eq_dec les les0).
      × left.
        unfold labelleq.
        right.
        eapply ex_intro.
        eapply ex_intro.
        split; eauto.
      × right.
        intro.
        destruct n.
        unfold labelleq in H.
        destruct H.
        inversion H.
        destruct H as [les1 [les2 [H1 [H2 H3]]]].
        inversion H1.
        inversion H2.
        subst.
        auto.
Qed.

Lemma labeleq_dec:
  ∀ l1 l2,
    {labeleq l1 l2} + {¬ labeleq l1 l2}.
Proof.
  intros.
  destruct (labelleq_dec l1 l2); destruct (labelleq_dec l2 l1);
    try solve [
          left;
          split; auto];
    try solve [
          right;
          intro Hlabeleq; auto;
          destruct Hlabeleq;
          auto].
Qed.

Lemma labeleq_trans :
