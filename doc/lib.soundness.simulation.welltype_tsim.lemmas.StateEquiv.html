Lemma PStoreApprox_trans :
  ∀ ps1 ps2 ps3 l envs
    (HPStoreApprox12: PStoreApprox ps1 ps2 l envs)
    (HPStoreApprox23: PStoreApprox ps2 ps3 l envs),
    (PStoreApprox ps1 ps3 l envs).
Proof.
  unfold PStoreApprox.
  intros.
  rename ts2 into ts3.
  rename n2 into n3.
  rename Hps2x into Hps3x.
  remember (ps2 x) as v.
  rename Heqv into Hps2x.
  destruct v as [ts2 n2].
  apply eq_sym in Hps2x.
  assert (n1 = n2 ∧ ts1 = ts2) as Heq12 by (eapply HPStoreApprox12; eauto).
  assert (n2 = n3 ∧ ts2 = ts3) as Heq23 by (eapply HPStoreApprox23; eauto).
  crush.
Qed.

Hint Resolve PStoreApprox_trans: basestate.

Lemma PStoreApprox_lt_envp:
  ∀ ps ps' l envp1 envp2
  (Henvleq : leq_env envp1 envp2)
  (HPStoreApprox: PStoreApprox ps ps' l envp1),
  PStoreApprox ps ps' l envp2.
Proof.
  intros.
  unfold PStoreApprox in ×.
  intros.
  eapply HPStoreApprox; eauto.
  subst.
  eauto with baselabel; eauto.
Qed.

Lemma PStateApprox_leq_env:
  ∀ pst pst' l pc envp envp'
    (Hleqenv: leq_env envp envp')
    (HPStateApprox: PStateApprox pst pst' l pc envp),
    PStateApprox pst pst' l pc envp'.
Proof.
  intros.
  destruct pst as [tpc ps].
  destruct pst' as [tpc' ps'].
  destruct HPStateApprox as [HPStateApprox1 HPStateApprox2].
  split.
  - auto.
  - unfold PStoreApprox in ×.
    intros.
    eapply HPStateApprox2; eauto.
    unfold leq_env in Hleqenv.
    subst.
    eauto with baselabel.
Qed.

Hint Resolve PStateApprox_leq_env: baseenv.
