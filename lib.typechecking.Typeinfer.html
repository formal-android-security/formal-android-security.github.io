The Environment


Definition Env := Vars → Label.

Definition leq_env (env1 env2 : Env) := ∀ x, labelleq (env1 x) (env2 x).

Definition env_union (env1 env2 : Env) :=
  fun x ⇒ match (env1 x, env2 x) with
           | (l1,l2) ⇒ lunion l1 l2
           end.

Definition updateenv (env:Env) (x:Vars) (l:Label) :=
  fun y ⇒
    if eq_nat_dec x y
    then l
    else env(y).

TypeSafe for Expression

Inductive ExprTypeSafe : Env → Expr → Label → Prop :=
| ETSVar : ∀ envp x l,
    envp(x) = l →
    ExprTypeSafe envp (EVar x) l
| ETSNum : ∀ envp n,
    ExprTypeSafe envp (ENum n) lempty
| ETSAdd : ∀ envp e1 e2 l1 l2,
    ExprTypeSafe envp e1 l1 →
    ExprTypeSafe envp e2 l2 →
    ExprTypeSafe envp (EAdd e1 e2) (lunion l1 l2).

TypeSafe for Bool Expression

Inductive BexpTypeSafe : Env → Bexp → Label → Prop :=
| BTStrue : ∀ envp, BexpTypeSafe envp Btrue lempty
| BTSfalse : ∀ envp, BexpTypeSafe envp Bfalse lempty
| BTSBeq : ∀ envp e1 e2 l1 l2,
    ExprTypeSafe envp e1 l1 →
    ExprTypeSafe envp e2 l2 →
    BexpTypeSafe envp (Beq e1 e2) (lunion l1 l2)
| BTSBleq : ∀ envp e1 e2 l1 l2,
    ExprTypeSafe envp e1 l1 →
    ExprTypeSafe envp e2 l2 →
    BexpTypeSafe envp (Bleq e1 e2) (lunion l1 l2)
| BTSBnot : ∀ envp b l,
    BexpTypeSafe envp b l →
    BexpTypeSafe envp (Bnot b) l
.

TypeSafe for Command

Inductive TypeSafe : (Label × Policy × Env) → Env → Cmd → Env → Prop:=
| TSSkip : ∀ pc po envs envp,
    TypeSafe (pc,po,envs) envp Skip envp
| TSHrecv : ∀ pc gap gsp envs envp x y envp',
    envp' = updateenv (updateenv envp x (lunion gap pc)) y (lunion gap pc) →
    (labelleq pc (gap)) →
    TypeSafe (pc,(gap,gsp),envs) envp (Hrecv x y) envp'
| TSLrecv : ∀ pc po envs envp x y envp',
    envp' = updateenv (updateenv envp x pc) y pc →
    pc = lempty →
    TypeSafe (pc,po,envs) envp (Lrecv x y) envp'
| TSHsend : ∀ pc po envs envp e1 e2,
    TypeSafe (pc,po,envs) envp (Hsend e1 e2) envp
| TSLsend : ∀ pc po envs envp e1 e2,
    ExprTypeSafe envp e1 lempty →
    ExprTypeSafe envp e2 lempty →
    pc = lempty →
    TypeSafe (pc,po,envs) envp (Lsend e1 e2) envp
| TSAssign : ∀ pc po envs envp x e l envp',
    ExprTypeSafe envp e l →
    envp' = updateenv envp x (lunion pc l) →
    TypeSafe (pc,po,envs) envp (Assign x e) envp'
| TSGet : ∀ pc gap gsp envs envp x y l envp',
    ExprTypeSafe envs (EVar y) l →
    labelleq l gap →
    envp' = updateenv envp x (lunion pc l) →
    TypeSafe (pc,(gap,gsp),envs) envp (Get x y) envp'
| TSPut : ∀ pc gap gsp envs envp x e l lx,
    ExprTypeSafe envp e l →
    ExprTypeSafe envs (EVar x) lx →
    labelleq (lunion l pc) lx→
    labelleq l gap →
    TypeSafe (pc,(gap,gsp),envs) envp (Put x e) envp
| TSOut : ∀ pc gap gsp envs e l envp,
    ExprTypeSafe envp e l →
    labelleq (lunion pc l) gsp →
    TypeSafe (pc,(gap,gsp),envs) envp (Out e) envp
| TSSeq : ∀ pc po envs envp c1 envp' c2 envp'',
    TypeSafe (pc,po,envs) envp c1 envp' →
    TypeSafe (pc,po,envs) envp' c2 envp'' →
    TypeSafe (pc,po,envs) envp (Seq c1 c2) envp''
| TSIfelse : ∀ pc po envs envp envp' b c1 c2 l,
    BexpTypeSafe envp b l →
    TypeSafe ((lunion pc l),po,envs) envp c1 envp' →
    TypeSafe ((lunion pc l),po,envs) envp c2 envp'→
    TypeSafe (pc,po,envs) envp (Ifelse b c1 c2) envp'
| TSWhiledo : ∀ pc po envs envp b c l,
    BexpTypeSafe envp b l →
    TypeSafe (lunion pc l,po,envs) envp c envp →
    TypeSafe (pc,po,envs) envp (Whiledo b c) envp.

The Translation


Inductive Trans : (Cmd × Label × Policy × Env × Env) → (AnnoCmd × Env) → Prop :=
| TSkip : ∀ pc po envs envp envp'
                 (HTypeSafe: TypeSafe (pc, po, envs) envp Skip envp'),
    Trans (Skip, pc, po , envs, envp) (ASkip, envp')
| TAssign : ∀ x e pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Assign x e) envp' →
    Trans (Assign x e, pc, po, envs, envp) (AAssign x e, envp')
| TGet : ∀ x y pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Get x y) envp' →
    Trans (Get x y, pc, po, envs, envp) (AGet x y, envp')
| TPut: ∀ x e pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Put x e) envp' →
    Trans (Put x e, pc, po, envs, envp) (APut x e, envp')
| TOut: ∀ e pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Out e) envp' →
    Trans (Out e, pc, po, envs, envp) (AOut e, envp')
| TSeq: ∀ c1 c2 pc po envs envp ac1 ac2 envp' envp'',
    Trans (c1, pc, po, envs, envp) (ac1,envp') →
    Trans (c2, pc, po, envs, envp') (ac2,envp'') →
    Trans (Seq c1 c2, pc, po, envs, envp) (ASeq ac1 ac2,envp'')
| TIf: ∀ b c1 c2 pc po envs envp ac1 ac2 l envp1 envp2,
    BexpTypeSafe envp b l →
    Trans (c1, lunion pc l, po, envs, envp) (ac1,envp1) →
    Trans (c2, lunion pc l, po, envs, envp) (ac2,envp2) →
    Trans (Ifelse b c1 c2, pc, po, envs, envp) (ASeq (AIfelse b ac1 ac2) AEnd, env_union envp1 envp2)
| TWhiledo0: ∀ b c pc po envs envp ac l envp',
    BexpTypeSafe envp b l →
    Trans (c, lunion pc l, po, envs, envp) (ac,envp') →
    envp = envp' →
    Trans (Whiledo b c, pc, po, envs, envp) (AWhiledo b (ASeq ac AEnd), envp')
| TWhiledon: ∀ b c pc po envs envp ac l envp' ac' envp'',
    BexpTypeSafe envp b l →
    Trans (c, lunion pc l, po, envs, envp) (ac',envp'') →
    envp ≠ envp'' →
    Trans (Whiledo b c, pc, po, envs, env_union envp envp'') (ac, envp') →
    Trans (Whiledo b c, pc, po, envs, envp) (ac, envp')
| THsend : ∀ e1 e2 pc (po:Policy) envs envp po l1 l2,
    ExprTypeSafe envp e1 l1→
    ExprTypeSafe envp e2 l2→
    Trans (Hsend e1 e2, pc, po, envs, envp) (AHsend e1 e2 (lunion (lunion pc l1) l2), envp)
| TLsend : ∀ e1 e2 pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Lsend e1 e2) envp' →
    Trans (Lsend e1 e2, pc, po, envs, envp) (ALsend e1 e2, envp')
| THrecv : ∀ x y pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Hrecv x y) envp' →
    Trans (Hrecv x y, pc, po, envs, envp) (AHrecv x y, envp')
| TLrecv : ∀ x y pc po envs envp envp',
    TypeSafe (pc,po,envs) envp (Lrecv x y) envp' →
    Trans (Lrecv x y, pc, po, envs, envp) (ALrecv x y, envp')
.

Definition Env_Map:= Tid → Env.

Definition PC_Map := Tid → Label.

Definition PolicysFromProg (p: Prog) (rho: Policys):=
  ∀ i po c, ProgMod.get p i = Some (po,c) → rho i = po.

Inductive TransProg : (Prog × PC_Map × Env × Env_Map) → (Policys × Progexe) → Prop :=
| TProg:
    ∀ p pcs envs envps rho pe envps',
      (∀ c po t ac
              (Hprogget: ProgMod.get p t = Some (po, c))
              (Hpeget: PEMod.get pe t = Some ac),
          Trans (c, pcs t, rho t, envs, envps t) (ac, envps' t))
      → (PolicysFromProg p rho)
      → TransProg (p, pcs, envs, envps) (rho, pe).
