Lemma peSkip_isPSkip:
  ∀ pe,
    pe = peSkip pe → isPSkip pe.
Proof.
  intros pe Hpe.
  unfold isPSkip.
  intros i ac Hpeget.
  rewrite Hpe in Hpeget.
  unfold peSkip in ×.
  rewrite PEMod.mapmap_sem in Hpeget.
  destruct (PEMod.get pe i); congruence.
Qed.

Lemma ParaStep_peSkip_eq:
  ∀ pe pe' rho sigma sigma'
         (HParaStep : ParaStep rho (pe, sigma) (pe', sigma')),
    peSkip pe = peSkip pe'.
Proof.
  intros.
  inversion HParaStep.
  subst.
  unfold peSkip.
  eapply PEMod.extensionality.
  intros t.
  rewrite PEMod.mapmap_sem.
  rewrite PEMod.mapmap_sem.
  unfold updatepe.
  destruct (PEMod.eq_dec i t).
  + subst t.
    rewrite PEMod.set_a_get_a; eauto with basemap.
    match goal with [H: PEMod.get _ _ = Some _ |- _] ⇒ rewrite H end.
    auto.
  + rewrite PEMod.set_a_get_a'; eauto with basemap.
Qed.

Lemma kParaStep_peSkip_eq:
  ∀ k pe pe' rho sigma sigma'
         (HkParaStep : kParaStep k rho (pe, sigma) (pe', sigma')),
    peSkip pe = peSkip pe'.
Proof.
  induction k.
  - intros.
    simpl in ×.
    congruence.
  - intros.
    simpl in ×.
    destruct HkParaStep as [sc'' [HParaStep HkParaStep]].
    destruct sc'' as [pe1' [psts1' sst1']].
    assert (peSkip pe1' = peSkip pe'). {
      eapply IHk; eauto.
    }
    assert (peSkip pe = peSkip pe1'). {
      eapply ParaStep_peSkip_eq; eauto.
    }
    congruence.
Qed.

Lemma starParaStep_peSkip_eq:
  ∀ pe pe' rho sigma sigma'
         (HstarParaStep : starParaStep rho (pe, sigma) (pe', sigma')),
    peSkip pe = peSkip pe'.
Proof.
  intros.
  destruct HstarParaStep as [k HkParaStep].
  eapply kParaStep_peSkip_eq; eauto.
Qed.


Lemma PSim_Skip_SStateApprox_kParaStep:
  ∀ k pe pe' psts psts' sst sst' l pcs envps' rho envs i psts1 sst1
    (HPSim : PSim pe pe' psts psts' l pcs envps' rho envs i)
  
  
  (HkParaStep : kParaStep k rho (pe, (psts, sst))
                       (peSkip pe, (psts1, sst1)))
  (HSStateApprox : SharedStateApprox sst sst' i l rho envs)
  (HstateII: stateII (psts, sst) rho)
  (HstateII': stateII (psts', sst') rho)
  ,
  ∃
    psts2 sst2,
    starParaStep rho (pe', (psts', sst')) (peSkip pe', (psts2, sst2)) ∧
    SharedStateApprox sst1 sst2 i l rho envs.
Proof.
  induction k.
  - intros.
    simpl in ×.
    inversion HkParaStep.
    destruct HPSim as [HPSim1 HPSim2].
    destruct HPSim1 as [HPSim1 HPSim2].
    assert (isPSkip pe) as HPSkip. {
      eapply peSkip_isPSkip; eauto.
    }
    subst psts1 sst1.
    specialize (HPSim2 sst sst' HPSkip HSStateApprox).
    destruct HPSim2 as [psts1' [sst1' [HstarParaStep [HSStateApprox' [HPStateApprox [HpstI HpstI']]]]]]; auto.
    repeat eapply ex_intro.
    split; eauto.
  - intros.
    simpl in ×.
    destruct HkParaStep as [sc'' [HParaStep HkParaStep]].
    destruct sc'' as [pe1' [psts1' sst1']].
    destruct HPSim as [HPSim1 HPSim2].
    destruct HPSim1 as [HPSim1 HPSim2].
    specialize (HPSim1 sst sst' pe1' psts1' sst1' HSStateApprox HParaStep).
    destruct HPSim1 as [pe1'' [psts1'' [sst1'' [HstarParaStep [HSStateApprox'' [HPSim'' [HstateII1 HstateII1']]]]]]]; auto.
    assert (peSkip pe = peSkip pe1') as HpeSkip. {
      eapply ParaStep_peSkip_eq; eauto.
    }
    rewrite HpeSkip in HkParaStep.
    assert (∃ (psts2 : PrivateStateSet) (sst2 : SharedState),
               starParaStep rho (pe1'', (psts1'', sst1'')) (peSkip pe1'', (psts2, sst2)) ∧
               SharedStateApprox sst1 sst2 i l rho envs) as HstarparaSeqStep'. {
      eapply IHk with (pe:=pe1') (pe':=pe1''); eauto.
    }
    destruct HstarparaSeqStep' as [psts2 [sst2 [HstarParaStep' HSStateApprox']]].
    ∃ psts2, sst2.
    split; auto.
    assert (peSkip pe' = peSkip pe1'') as HpeSkip'. {
      eapply starParaStep_peSkip_eq; eauto.
    }
    rewrite <- HpeSkip' in HstarParaStep'.
    eapply starParaStep_starParaStep_starParaStep; eauto.
Qed.

Lemma PSim_Skip_SStateApprox_starParaStep:
  ∀ pe pe' psts psts' sst sst' l pcs envps' rho envs i psts1 sst1
    (HPSim : PSim pe pe' psts psts' l pcs envps' rho envs i)
  
  
  (HstarParaStep : starParaStep rho (pe, (psts, sst))
                       (peSkip pe, (psts1, sst1)))
  (HSStateApprox : SharedStateApprox sst sst' i l rho envs)
  (HstateII: stateII (psts, sst) rho)
  (HstateII': stateII (psts', sst') rho),
  ∃
    psts2 sst2,
    starParaStep rho (pe', (psts', sst')) (peSkip pe', (psts2, sst2)) ∧
    SharedStateApprox sst1 sst2 i l rho envs.
Proof.
  intros.
  destruct HstarParaStep as [k HkParaStep].
  eapply PSim_Skip_SStateApprox_kParaStep; eauto.
Qed.
