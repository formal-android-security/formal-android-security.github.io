Termination about SeqStep

Lemma SeqStep_TTerm:
  ∀ t rho ac pst sst ac' pst' sst'
    (HSeqStep: SeqStep t rho (ac, (pst, sst)) (ac', (pst',sst')))
    (HTTerm: ∀ sst'', ThreadTerm t rho ac pst sst''),
    ∀ sst'', ThreadTerm t rho ac' pst' sst''.
Proof.
  intros.
  generalize HTTerm; intros HTTerm_aux.
  specialize (HTTerm sst).
  induction HTTerm.
  subst.
  destruct pst.
  inversion HSeqStep.
  subst.
  constructor; auto.

  subst.
  destruct (eq_AnnoCmd_dec ac ac').
  + assert ((pst,sst) = (pst', sst')) as Heqv
        by (eapply SeqStep_ac_eq_seqstate_eq; eauto).
    inversion Heqv.
    subst ac' pst' sst'.
    intros.
    eauto.
  + destruct (eq_AnnoCmd_dec ac ac'0).
    × assert ((pst,sst) = (pst'0, sst'0)) as Heqv
          by (eapply SeqStep_ac_eq_seqstate_eq; eauto).
      inversion Heqv.
      subst ac'0 pst'0 sst'0.
      eapply H1; eauto.
    × assert (ac' = ac'0)
        by (eapply SeqStep_determine_ac; eauto).
      assert ((pst', sst') = (pst'0, sst'0)) as Heqv
          by (eapply SeqStep_determine_seqstate; eauto).
      inversion Heqv.
      subst.
      eauto.
Qed.

Lemma kSeqStep_TTerm:
  ∀ k t rho ac pst sst ac' pst' sst'
    (HkSeqStep: kSeqStep k t rho (ac, (pst, sst)) (ac', (pst',sst')))
    (HTTerm: ∀ sst'0, ThreadTerm t rho ac pst sst'0),
    ∀ sst'0, ThreadTerm t rho ac' pst' sst'0.
Proof.
  induction k.
  - intros.
    simpl in ×.
    inversion HkSeqStep.
    subst; auto.
  - intros.
    simpl in ×.
    destruct HkSeqStep as [sc'' [HSeqStep HkSeqStep]].
    destruct sc'' as [ac'' [pst'' sst'']].
    eapply IHk; eauto.
    eapply SeqStep_TTerm; eauto.
Qed.

Lemma Seq_TTerm_TTerm:
  ∀ t rho ac1 ac2 pst sst
         (HTTerm: ThreadTerm t rho (ASeq ac1 ac2) pst sst),
    ThreadTerm t rho ac1 pst sst.
Proof.
  intros.
  remember (fun t' rho' ac' pst' sst' ⇒
              match ac' with
              | ASeq ac1' ac2' ⇒
                
                ThreadTerm t' rho' ac1' pst' sst'
              | _ ⇒ True
              end)
           as P.
  assert (P t rho (ASeq ac1 ac2) pst sst = ThreadTerm t rho ac1 pst sst) as HP.
  {
    subst; auto.
  }
  rewrite <- HP; clear HP.
  induction HTTerm.
  - subst.
    auto.
  - subst.
    destruct ac; auto.
    inversion H.
    + subst.
      econstructor; eauto.
    + subst.
      constructor; auto.
Qed.

Lemma Seq_TTerm_TTerm':
  ∀ t rho ac1 ac2 pst sst pst1 sst1
         (HstarSeqStep : starSeqStep t rho (ac1, (pst, sst)) (ASkip, (pst1, sst1)))
         (HTTerm: ∀ sst'0, ThreadTerm t rho (ASeq ac1 ac2) pst sst'0),
    ∀ sst1'0, ThreadTerm t rho ac2 pst1 sst1'0.
Proof.
  intros.
  eapply starSeqStep_TTerm with (ac:=ASeq ac1 ac2); eauto.
  - assert (starSeqStep t rho (ASeq ac1 ac2, (pst, sst)) (ASeq ASkip ac2, (pst1, sst1))). {
      eapply starSeqStep_ASeq_starSeqStep; eauto.
    }
    assert (starSeqStep t rho (ASeq ASkip ac2, (pst1, sst1)) (ac2, (pst1, sst1))). {
      ∃ (S O).
      simpl.
      eapply ex_intro; eauto.
      split; eauto.
      econstructor; eauto.
    }
    eapply starSeqStep_trans; eauto.
Qed.

Lemma TTerm_starSeqStep_Skip:
  ∀ t rho ac pst sst
    (HTTerm: ThreadTerm t rho ac pst sst),
    ∃ pst' sst', starSeqStep t rho (ac, (pst, sst)) (ASkip, (pst',sst')).
Proof.
  intros.
  induction HTTerm.
  - ∃ pst, sst, O.
    simpl.
    auto.
  - match goal with [H: ∀ _, ∃ _ _, starSeqStep _ _ _ _ |- _] ⇒ rename H into HstarSeqStep end.
    specialize (HstarSeqStep sst').
    destruct HstarSeqStep as [pst'0 [sst'0 HstarSeqStep]].
    destruct HstarSeqStep as [k HkSeqStep].
    ∃ pst'0, sst'0.
    ∃ (S k).
    simpl.
    ∃ (ac', (pst', sst')).
    split; auto.
Qed.

Termination about ParaStep

Lemma ParaStep_PTerm:
  ∀ rho pe psts sst pe' psts' sst'
    (HParaStep: ParaStep rho (pe, (psts, sst)) (pe', (psts',sst')))
    (HPTerm: ProgTerm rho pe psts),
    ProgTerm rho pe' psts'.
Proof.
  intros.
  inversion HParaStep; clear HParaStep.
  subst.
  match goal with [H: PSTSMod.get _ _ = Some _ |- _] ⇒ rename H into Hgetpsts end.
  match goal with [H: PEMod.get _ _ = Some _ |- _] ⇒ rename H into Hgetpe end.
  unfold ProgTerm.
  intros.
  destruct (PEMod.eq_dec i t).
  - subst i.
    unfold updatepsts in ×.
    unfold updatepe in ×.
    rewrite PSTSMod.set_a_get_a in *; auto with basemap.
    rewrite PEMod.set_a_get_a in *; auto with basemap.
    inversion Hgetpsts0; clear Hgetpsts0.
    inversion Hgetpe0; clear Hgetpe0.
    subst pst0 ac.
    unfold ProgTerm in HPTerm.
    specialize (HPTerm t aci pst Hgetpsts Hgetpe).
    eapply SeqStep_TTerm; eauto.
  - unfold updatepsts in ×.
    unfold updatepe in ×.
    rewrite PSTSMod.set_a_get_a' in *; auto with basemap.
    rewrite PEMod.set_a_get_a' in *; auto with basemap.
Qed.
